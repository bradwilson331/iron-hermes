```rust
/// Top-level desktop/web shell composer.
///
/// Port of `warp2ironhermes/project/app/app.jsx` `WarpHermes` (lines 119-275)
/// per CONTEXT D-01. Phase 3 is pure-static (D-06): all values are hardcoded
/// or come from `demo_*()` fixtures in `src/state.rs`. The four data-attributes
/// are baked at the prototype defaults (UI-SPEC lines 491-510); Phase 5

/// visibility); the InputBox is rendered with `focused: true` per UI-SPEC
/// planner-handoff #4 so the focus ring is visible during Phase 3 UAT.
#[component]
pub fn WarpHermes() -> Element {
    let blocks = demo_blocks();
    let messages = demo_messages();
    let palette_items = demo_palette_items();
    let tabs = demo_tabs();
    let tokens = TokenBudget { used: 12_300, max: 128_000 };

    rsx! {
        div {
            class: "wh-app",
            "data-theme": "cyan",
            "data-density": "comfy",
            "data-block": "framed",
            "data-agent": "right",
            TitleBar { tabs: tabs, active_tab: 0_usize, show_traffic_lights: true }
            div { class: "wh-main",
                div { class: "wh-col",
                    BlockStream { blocks: blocks }
                    InputBox { mode: Mode::Shell, focused: true }
                    StatusBar {
                        mode: "Chat".to_string(),
                        model: "claude-sonnet-4".to_string(),
                        provider: "anthropic".to_string(),
                        tokens: tokens,
                        scanner_active: true,
                        hint: "/help · ⌃C cancel · ⌘K palette".to_string(),
                    }
                }
                AgentPanel { messages: messages, personality: "default".to_string() }
            }
            CommandPalette { items: palette_items, query: String::new(), open: true }
        }
    }
}
```
