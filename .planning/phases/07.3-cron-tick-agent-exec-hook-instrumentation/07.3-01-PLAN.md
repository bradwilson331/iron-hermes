// --- 10. Cron tick task ---
if let Some(ref job_store) = self.job_store {
    let tick_cancel = self.cancel.clone();
    let job_store_tick = job_store.clone();
    let skill_registry_tick = self.skill_registry.clone();
    join_set.spawn(async move {
        let mut interval = tokio::time::interval(tokio::time::Duration::from_secs(60));
        interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
        loop {
            tokio::select! {
                _ = tick_cancel.cancelled() => { info!("Cron tick task shutting down"); break; }
                _ = interval.tick() => {
                    match ironhermes_cron::run_tick_check(&job_store_tick).await {
                        Ok((due_jobs, result, _lock_guard)) => {
                            if result.jobs_run > 0 { info!("Tick: {} jobs due", due_jobs.len()); }
                            for job in &due_jobs {
                                info!("Job due: {} ({})", job.name, job.id);
                                let _full_input = if let Some(ref skill_reg) = skill_registry_tick {
                                    let skill_context = resolve_skill_context(skill_reg, &job.skills);
                                    if skill_context.is_empty() { job.prompt.clone() }
                                    else { format!("{}\n\n---\n\n{}", skill_context, job.prompt) }
                                } else { job.prompt.clone() };
                                // Mark as run with placeholder output
                                // Full agent execution requires AgentLoop integration
                                // which involves building a fresh agent per job
                                if let Err(e) = ironhermes_cron::complete_job_run(
                                    &job_store_tick, job,
                                    "[Tick runner: agent execution pending full integration]",
                                    true
                                ).await {
                                    error!("Failed to complete job run: {}", e);
                                }
                            }
                        }
                        Err(e) => { error!("Tick error: {}", e); }
                    }
                }
            }
        }
    });
    info!("Cron tick task started (60s interval)");
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Cron scheduler -> AgentLoop | Scheduled job prompts + attached skill content cross into the LLM system prompt. Skill content is user-controlled via `~/.ironhermes/skills/` so it inherits the same injection-surface concerns as the Telegram path. |
| AgentLoop -> ToolRegistry::dispatch | Tool calls issued by the cron-triggered agent cross the existing tool guardrail chain — SAME chain as Telegram path, no new ingress. |
| AgentLoop -> HookRegistry listeners | Hook events (JSONL sink, webhook sink) cross into external sinks. Content previews are truncated via `preview(&content, 200)`. |
| LLM error -> complete_job_run -> delivery sinks | Raw LLM error strings must NOT reach JSONL / webhooks. Sanitized via `format!("[Agent error: {}]", e)` which emits only the `Display` impl of the error. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07.3-01 | Elevation of Privilege | execute_cron_job -> AgentLoop -> ToolRegistry::dispatch | mitigate | The cron path uses the SAME `Arc<ToolRegistry>` as Telegram (`self.tool_registry.clone()` in Task 4 Sub-step 4c). Guardrail chain runs at tool dispatch time inside `AgentLoop::execute_tool_call()`, not at prompt-construction time, so cron-injected skill content cannot bypass it. Verification: `runner.rs` grep confirms `tool_registry_tick = self.tool_registry.clone()` and `AgentLoop::new(client, tool_registry_tick.clone(), max_turns)` — same Arc flows into both paths. |
| T-07.3-02 | Information Disclosure | JSONL hook sink / webhook sink log injection via `chat_id` | accept | `chat_id` is derived from `job.origin.chat_id` (set server-side by `TelegramTool::cronjob_create` — user-supplied as Telegram chat ID, which is a numeric string validated by Telegram) or falls back to `job.id` (UUID-v4 generated server-side via `uuid::Uuid::new_v4()`). Neither path allows user-controlled arbitrary strings into the log sink's `chat_id` field. Residual risk: a malicious Telegram user can influence `chat_id` to be their own chat ID — this is by design (they are the owner of that chat). No mitigation required. |
| T-07.3-03 | Information Disclosure | Raw LLM error -> complete_job_run -> JSONL / webhook | mitigate | Task 4 Sub-step 4b wraps errors as `format!("[Agent error: {}]", e)` — this uses only `Display::fmt` on the `anyhow::Error`, which by convention does NOT include internal prompt fragments or raw LLM response bodies. The sanitized string is what flows into `complete_job_run`'s `output` parameter. Grep-verifiable: `runner.rs` contains `"[Agent error: {}]"` literal. Residual risk: if an upstream error chain includes a prompt fragment in its `Display` message, it leaks. Accept residual — this matches the existing Telegram path's error handling (handler.rs:414-421 also uses plain error display). |
| T-07.3-04 | Denial of Service | Panicking AgentLoop -> crashed tick task -> silent disable of all scheduled jobs | mitigate | Task 4 Sub-step 4b returns `Result<()>` from `execute_cron_job`. The tick task closure (Sub-step 4c) wraps the call in `if let Err(e) = execute_cron_job(...).await { error!(...); }` — a single job failure logs and continues to the next job. `AgentLoop::run()` is documented to return `Result<AgentResult>` rather than panicking on transport failures (verified via agent_loop.rs:114 signature). Residual risk: an unanticipated panic inside AgentLoop crashes the current job's future, but the `join_set.spawn` wrapper catches panics via `tokio::task::JoinHandle` — the next tick still runs. Acceptable. No `catch_unwind` needed. |
