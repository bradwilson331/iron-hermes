// crates/ironhermes-hub/src/github.rs:182-209
async fn rate_limit_error(&self, resp: reqwest::Response) -> HubError {
    let remaining = resp
        .headers()
        .get("X-RateLimit-Remaining")
        .and_then(|v| v.to_str().ok())
        .and_then(|s| s.parse::<u64>().ok())
        .unwrap_or(0);

    let retry_after_s = if remaining == 0 {
        resp.headers()
            .get("X-RateLimit-Reset")
            .and_then(|v| v.to_str().ok())
            .and_then(|s| s.parse::<i64>().ok())
            .map(|reset_ts| {
                let now = chrono::Utc::now().timestamp();
                (reset_ts - now).max(0) as u64
            })
    } else {
        None
    };

    HubError::Typed {
        kind: HubErrorKind::RateLimited,
        message: "GitHub API rate limit exceeded".to_string(),
        suggestion: Some("Set HERMES_GITHUB_TOKEN or wait for rate-limit window to reset".to_string()),
        retry_after_s,
    }
}
```

**Replicate** as-is for the Trees API hop. **Skip** for the raw and skills.sh hops — those don't have GitHub-style rate-limit headers.

impl HubManifest {
    pub fn load_or_default() -> anyhow::Result<Self> {
        let p = crate::paths::manifest_path()?;
        if !p.exists() {
            return Ok(Self::default());
        }
        Ok(serde_json::from_str(&std::fs::read_to_string(p)?)?)
    }

    pub fn save(&self) -> anyhow::Result<()> {
        let p = crate::paths::manifest_path()?;
        if let Some(parent) = p.parent() {
            std::fs::create_dir_all(parent)?;
        }
        let tmp = p.with_extension("json.tmp");
        std::fs::write(&tmp, serde_json::to_string_pretty(self)?)?;
        std::fs::rename(tmp, p)?;
        Ok(())
    }
}

// crates/ironhermes-hub/src/tarball.rs:36-88
pub fn validate_bundle_rel_path(rel: &str) -> Result<String, HubError> {
    if rel.is_empty() {
        return Err(typed(HubErrorKind::Parse, "empty tar entry path"));
    }
    if rel.contains('\0') {
        return Err(typed(HubErrorKind::Parse, "NUL byte in tar entry path"));
    }
    let p = Path::new(rel);
    if p.is_absolute() {
        return Err(typed(
            HubErrorKind::Parse,
            format!("absolute tar entry path rejected: {rel}"),
        ));
    }
    for comp in p.components() {
        match comp {
            Component::ParentDir => {
                return Err(typed(
                    HubErrorKind::Parse,
                    format!(".. component rejected: {rel}"),
                ))
            }
            Component::Prefix(_) => {
                return Err(typed(
                    HubErrorKind::Parse,
                    format!("drive-letter/prefix rejected: {rel}"),
                ))
            }
            Component::RootDir => {
                return Err(typed(
                    HubErrorKind::Parse,
                    format!("root component rejected: {rel}"),
                ))
            }
            Component::CurDir | Component::Normal(_) => {}
        }
    }
    // ...
}
```

**Replicate** the `std::path::Component` match idiom for `sanitize_subpath`, but emit `HubErrorKind::PathTraversal` (new variant — see error.rs section) instead of `Parse`. **Different:** `sanitize_subpath` normalizes `\` → `/` first (D-18, matches reference `source-parser.ts:89-105`); traversal, NUL, absolute, drive-prefix all return typed errors with `kind = PathTraversal`.

## Project-Wide Rust Conventions the Planner Should Flag

Spotted during analog extraction; these aren't per-file patterns but cross-cutting posture checks the planner should explicitly assert in each plan:

1. **Error type is `anyhow::Result` at the outer boundary, `Result<_, HubError>` at the inner boundary.** Observe: `manifest.rs::load_or_default -> anyhow::Result<Self>` (outer, cross-crate) vs `installer.rs::install -> Result<InstallOutcome, HubError>` (inner, same-crate).
