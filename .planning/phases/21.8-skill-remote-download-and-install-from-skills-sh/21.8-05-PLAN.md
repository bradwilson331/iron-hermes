        let _guard = ENV_LOCK.lock().unwrap_or_else(|e| e.into_inner());
        let prev: Vec<_> = vars.iter().map(|(k, _)| (*k, std::env::var(k).ok())).collect();
        unsafe {
            for (k, v) in &vars {
                match v {
                    Some(val) => std::env::set_var(k, val),
                    None => std::env::remove_var(k),
                }
            }
