    ```rust
        /// Phase 25.3 D-T-1 / Discretion D-2 override: redact secrets from URL-shaped args.
        ///
        /// The web_extract schema accepts a `urls: Vec<String>` argument. Each URL may
        /// contain query-parameter secrets (api_key, token, etc.). This override applies
        /// `redact_secrets_in_url` (Phase 25.2 Plan 16) to every URL before the AgentLoop
        /// records the args in `trajectories.jsonl`.
        ///
        /// Other arg fields (e.g., `summarize`, `tier`) are passed through unchanged.
        fn redact_args(&self, raw: &serde_json::Value) -> serde_json::Value {
            use crate::web_extract::sanitize::redact_secrets_in_url;
            let mut redacted = raw.clone();
            if let Some(obj) = redacted.as_object_mut() {
                if let Some(urls) = obj.get_mut("urls") {
                    if let Some(arr) = urls.as_array_mut() {
                        for u in arr.iter_mut() {
                            if let Some(s) = u.as_str() {
                                let redacted_url = redact_secrets_in_url(s, &[]);
                                *u = serde_json::Value::String(redacted_url);
                            }
                        }
                    } else if let Some(s) = urls.as_str() {
                        // Tolerate single-string url field too (some callers may pass scalar)
                        let redacted_url = redact_secrets_in_url(s, &[]);
                        *urls = serde_json::Value::String(redacted_url);
                    }
                }
            }
            redacted
        }
    ```
