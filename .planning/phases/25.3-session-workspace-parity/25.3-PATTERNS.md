```rust
use std::fs::{File, OpenOptions};
use std::io::Write;
use std::path::PathBuf;
use anyhow::Context as _;

pub struct TrajectoryWriter {
    file: File,
    session_id: String,
    path: PathBuf,
}

impl TrajectoryWriter {
    /// Open (or create) the trajectory file for the given session.
    /// Creates parent directories if needed — same pattern as StateStore::new.
    pub fn open(path: impl Into<PathBuf>) -> anyhow::Result<Self> {
        let path = path.into();
        if let Some(parent) = path.parent() {
            std::fs::create_dir_all(parent)
                .with_context(|| format!("create trajectory directory {}", parent.display()))?;
        }
        let file = OpenOptions::new()
            .create(true)
            .append(true)
            .open(&path)
            .with_context(|| format!("open trajectory file {}", path.display()))?;
        Ok(Self { file, session_id: String::new(), path })
    }

    /// Append one entry as a JSONL line + fsync.
    /// Crash-safe: sync_data() flushes kernel buffers to storage (Pitfall 3).
    pub fn append(&mut self, entry: &crate::format::TrajectoryEntry) -> anyhow::Result<()> {
        let line = serde_json::to_string(entry)? + "\n";
        self.file.write_all(line.as_bytes())?;
        self.file.sync_data()?;   // fsync per line — crash safety
        Ok(())
    }
}

/// Drop impl: final fsync on session end (Pitfall 3 guard).
impl Drop for TrajectoryWriter {
    fn drop(&mut self) {
        let _ = self.file.sync_data();   // .ok() semantics via let _
    }
}
```

**Target shape:** TrajectoryReader is consumed by 25.4's Curator. For 25.3, expose the minimal API needed:

```rust
use std::io::{BufRead, BufReader};
use std::path::PathBuf;
use anyhow::Context as _;

pub struct TrajectoryReader {
    path: PathBuf,
}

impl TrajectoryReader {
    pub fn open(path: impl Into<PathBuf>) -> Self {
        Self { path: path.into() }
    }

    /// Read all entries from the JSONL file. Returns empty Vec if file does not exist.
    pub fn read_all(&self) -> anyhow::Result<Vec<crate::format::TrajectoryEntry>> {
        if !self.path.exists() {
            return Ok(vec![]);
        }
        let file = std::fs::File::open(&self.path)
            .with_context(|| format!("open trajectory file {}", self.path.display()))?;
        let reader = BufReader::new(file);
        let mut entries = Vec::new();
        for line in reader.lines() {
            let line = line?;
            if line.trim().is_empty() { continue; }
            let entry: crate::format::TrajectoryEntry = serde_json::from_str(&line)?;
            entries.push(entry);
        }
        Ok(entries)
    }
}
```

```rust
use std::path::{Path, PathBuf};
use serde::{Deserialize, Serialize};

/// Per-cwd project workspace. Resolved once at session start (frozen-snapshot pattern,
/// Phase 17/18 precedent). Mid-session switching is explicitly deferred (D-W-1).
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Workspace {
    pub root: PathBuf,
    pub soul_path: Option<PathBuf>,
    pub agents_chain: Vec<PathBuf>,
    pub memory_dir: PathBuf,
    /// Write destination for Phase 25.4 Curator skill emission (D-C-3).
    pub skills_dir: PathBuf,
    pub tools_config: Option<PathBuf>,
}

/// Walk up from `cwd` looking for `.hermes/` or `.ironhermes/` directory.
/// Returns `Some(root)` when found; `None` means use global fallback (~/.ironhermes/).
///
/// Resolution order (D-W-1):
/// 1. Check each ancestor directory for `.hermes/` or `.ironhermes/`
/// 2. If found, build Workspace from that root
/// 3. If not found after reaching filesystem root, return None → caller uses get_hermes_home()
pub fn resolve_from_cwd(cwd: &Path) -> Option<Workspace> {
    let mut current = cwd.to_path_buf();
    loop {
        if current.join(".hermes").is_dir() || current.join(".ironhermes").is_dir() {
            let marker_dir = if current.join(".ironhermes").is_dir() {
                current.join(".ironhermes")
            } else {
                current.join(".hermes")
            };
            return Some(Workspace {
                soul_path: {
                    let p = current.join("SOUL.md");
                    if p.exists() { Some(p) } else { marker_dir.join("SOUL.md").exists().then(|| marker_dir.join("SOUL.md")) }
                },
                agents_chain: vec![],   // populated by scanning chain; planner fills
                memory_dir: marker_dir.join("memory"),
                skills_dir: current.join("skills"),
                tools_config: marker_dir.join("tools.yaml").exists().then(|| marker_dir.join("tools.yaml")),
                root: current,
            });
        }
        if !current.pop() {
            break;
        }
    }
    None
}
```

**Cache-stability note:** The `Workspace` struct must be resolved ONCE at session start and stored as `Arc<Workspace>`. Never re-resolve mid-session (Pitfall 2 / prompt cache breakage).

```rust
use std::path::{Path, PathBuf};
use anyhow::Context as _;

/// Write a session's data to the 4-file hermes-agent-compat directory layout.
///
/// Output: `<output_dir>/{messages.json, metadata.json, context.json, trajectories.jsonl}`
/// SQLite remains canonical; this is a derived on-demand export (D-F-1 hybrid).
pub struct SessionDirectoryExport {
    pub session_id: String,
    pub output_dir: PathBuf,
}

impl SessionDirectoryExport {
    pub fn write(
        &self,
        export: &SessionExport,                  // from StateStore::export_session
        context_json: Option<&str>,               // Phase 18 compressor output; None if not compressed
        trajectory_source: Option<&Path>,         // path to existing trajectories.jsonl; None if no writer ran
    ) -> anyhow::Result<()> {
        std::fs::create_dir_all(&self.output_dir)
            .with_context(|| format!("create export dir {}", self.output_dir.display()))?;

        // 1. messages.json — array of StoredMessage
        let messages_json = serde_json::to_string_pretty(&export.messages)?;
        std::fs::write(self.output_dir.join("messages.json"), messages_json)?;

        // 2. metadata.json — Session struct fields
        let metadata_json = serde_json::to_string_pretty(&export.session)?;
        std::fs::write(self.output_dir.join("metadata.json"), metadata_json)?;

        // 3. context.json — Phase 18 compressor output (empty object if None)
        let context_json = context_json.unwrap_or("{}");
        std::fs::write(self.output_dir.join("context.json"), context_json)?;

        // 4. trajectories.jsonl — copy or symlink from trajectory crate's path
        if let Some(src) = trajectory_source {
            if src.exists() {
                std::fs::copy(src, self.output_dir.join("trajectories.jsonl"))?;
            }
        }
        Ok(())
    }
}
```
