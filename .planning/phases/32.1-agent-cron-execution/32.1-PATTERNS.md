```rust
pub(crate) fn resolve_skill_context(
    registry: &ironhermes_core::SkillRegistry,
    skill_names: &[String],
) -> String {
    let mut parts = Vec::new();
    let mut skipped: Vec<String> = Vec::new();
    for name in skill_names {
        match registry.read_content(name) {
            Some(content) => parts.push(format!(
                "[IMPORTANT: The user has invoked the \"{}\" skill...]\n\n{}", name, content
            )),
            None => {
                tracing::warn!(skill = %name, "Skill not found at tick time - skipping");
                skipped.push(name.clone());
            }
        }
    }
    // skill-skip prefix (PARITY §2)
    let mut result = String::new();
    if !skipped.is_empty() {
        result.push_str(&format!(
            "[IMPORTANT: The following skill(s) were listed for this job but could not be \
             found and were skipped: {}.]\n\n", skipped.join(", ")
        ));
    }
    result.push_str(&parts.join("\n\n---\n\n"));
    result
}
```

**Tick loop scaffold pattern** (gateway runner.rs lines 751–832 — extract verbatim):
```rust
pub async fn run_tick_loop(
    ctx: Arc<CronRunnerContext>,
    cancel: tokio_util::sync::CancellationToken,
) {
    let mut interval = tokio::time::interval(tokio::time::Duration::from_secs(60));
    interval.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
    let mut first_tick = true;

    loop {
        tokio::select! {
            _ = cancel.cancelled() => {
                info!("Cron tick loop shutting down");
                break;
            }
            _ = interval.tick() => {
                if first_tick {
                    first_tick = false;
                    // first-tick burst guard (fast_forward_backlog)
                    continue;
                }
                // MCP discovery per tick (non-fatal)
                if let Some(ref mcp) = ctx.mcp_manager {
                    let _ = mcp.discover_tools().await; // TODO: verify actual method name
                }
                match ironhermes_cron::run_tick_check(&ctx.job_store).await {
                    Ok((due_jobs, result, _lock_guard)) => {
                        let (workdir_jobs, parallel_jobs): (Vec<_>, Vec<_>) =
                            due_jobs.iter().partition(|j| j.workdir.is_some());
                        // Sequential workdir jobs
                        for job in &workdir_jobs {
                            if let Err(e) = run_cron_job(job, &ctx).await {
                                error!("Job {} failed: {}", job.name, e);
                            }
                        }
                        // Parallel non-workdir jobs
                        let max_par = parse_env_usize("IRONHERMES_CRON_MAX_PARALLEL");
                        let mut set = tokio::task::JoinSet::new();
                        for job in parallel_jobs {
                            let ctx2 = ctx.clone();
                            let job = job.clone();
                            set.spawn(async move {
                                if let Err(e) = run_cron_job(&job, &ctx2).await {
                                    error!("Job {} failed: {}", job.name, e);
                                }
                            });
                            // if max_par is Some(n), acquire semaphore before spawn
                        }
                        while set.join_next().await.is_some() {}
                    }
                    Err(e) => error!("Tick error: {}", e),
                }
                // MCP orphan reaper after tick
            }
        }
    }
}
```

```rust
// activity_handle: Arc<Mutex<ActivitySummary>> shared with AgentLoop (new API)
pub async fn run_with_inactivity_timeout<F, T>(
    agent_future: F,
    activity_handle: Arc<tokio::sync::Mutex<ActivitySummary>>,
    inactivity_secs: u64,
) -> anyhow::Result<T>
where F: std::future::Future<Output = anyhow::Result<T>>
{
    if inactivity_secs == 0 {
        return agent_future.await;
    }
    tokio::select! {
        result = agent_future => result,
        _ = async {
            loop {
                tokio::time::sleep(Duration::from_secs(5)).await;
                let idle = activity_handle.lock().await.seconds_since;
                if idle > inactivity_secs as f64 {
                    break;
                }
            }
        } => Err(anyhow::anyhow!("inactivity timeout after {}s", inactivity_secs))
    }
}
```
