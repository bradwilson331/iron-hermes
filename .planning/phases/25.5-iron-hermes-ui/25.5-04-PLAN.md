   ```rust
   #[get("/api/ws/chat")]
   pub async fn ws_chat(ws: WebSocketOptions) -> Result<Websocket<String, String>, ServerFnError> {
       Ok(ws.on_upgrade(|mut socket| async move {
           while let Ok(msg) = socket.recv().await {
               // Parse incoming message
               let req: ChatRequest = match serde_json::from_str(&msg) {
                   Ok(r) => r,
                   Err(e) => {
                       let err = ChatStreamEvent::Error { message: e.to_string() };
                       let _ = socket.send(serde_json::to_string(&err).unwrap()).await;
                       continue;
                   }
               };

               // Build AgentLoop for this turn.
               // Create a channel for streaming events back to WebSocket.
               let (tx, mut rx) = tokio::sync::mpsc::unbounded_channel::<ChatStreamEvent>();

               // Construct the streaming callback that sends deltas to the channel
               let tx_clone = tx.clone();
               let stream_callback: Box<dyn Fn(&str) + Send + Sync> = Box::new(move |delta: &str| {
                   let _ = tx_clone.send(ChatStreamEvent::Delta { text: delta.to_string() });
               });

               // Spawn agent turn in background task
               let message = req.message.clone();
               let tx_fin = tx.clone();
               tokio::spawn(async move {
                   // Build AgentLoop (simplified — executor must adapt to actual API)
                   // Load config, build client, construct AgentLoop with stream callback
                   match build_and_run_agent_turn(message, stream_callback).await {
                       Ok(result) => {
                           let _ = tx_fin.send(ChatStreamEvent::Finished {
                               total_tokens: result.total_tokens.unwrap_or(0),
                           });
                       }
                       Err(e) => {
                           let _ = tx_fin.send(ChatStreamEvent::Error {
                               message: e.to_string(),
                           });
                       }
                   }
                   drop(tx_fin); // Signal channel closure
               });

               // Forward events from channel to WebSocket
               drop(tx); // Only the spawn task holds tx now
               while let Some(event) = rx.recv().await {
                   let json = serde_json::to_string(&event).unwrap_or_default();
                   if socket.send(json).await.is_err() {
                       break;
                   }
               }
           }
       }))
   }
   ```

   - Construct a ChatMessage from the user input

   - Return the AgentResult

   NOTE: The exact AgentLoop construction is complex (see main.rs run_chat for reference — it requires PromptBuilder, ContextEngine, etc.). The executor should study the actual `run_chat` function and extract the minimum viable setup. Start with the simplest case: a single-turn agent call with streaming.
