   use ironhermes_tools::delegate_task::{SubagentRunner, DelegateTaskTool};
   use std::sync::Arc;
   use std::time::Duration;
   use tokio_util::sync::CancellationToken;

   // Mock runner that sleeps and reports whether its cancel token was cancelled.
   struct SleepRunner {
       observed_cancelled: Arc<std::sync::atomic::AtomicBool>,
   }
   #[async_trait::async_trait]
   impl SubagentRunner for SleepRunner {
       async fn run_child(
           &self,
           _registry: std::sync::Arc<ironhermes_tools::registry::ToolRegistry>,
           _prompt: String,
           _max_iterations: usize,
           _model: Option<String>,
           cancel: Option<CancellationToken>,
           _cb: Option<ironhermes_tools::delegate_task::ChildToolProgressCallback>,
       ) -> anyhow::Result<String> {
           let observed = self.observed_cancelled.clone();
           let cancel = cancel.expect("cancel token required for this test");
           tokio::select! {
               _ = tokio::time::sleep(Duration::from_secs(10)) => Ok("done".into()),
               _ = cancel.cancelled() => {
                   observed.store(true, std::sync::atomic::Ordering::SeqCst);
                   anyhow::bail!("cancelled")
               }
           }
       }
   }

           timeout_secs: 1,
           max_iterations: 5,
           max_subagents: 2,
           // ...any other fields default
           ..Default::default()
       };
