```rust
//! Session-scoped in-memory background-process registry (D-23..D-29).

use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::{broadcast, watch, Mutex, RwLock};

pub const MAX_OUTPUT_CHARS: usize = 200_000;           // 200KB rolling buffer per process
pub const MAX_PROCESSES: usize = 64;                   // LRU-prune threshold
pub const FINISHED_TTL_SECONDS: u64 = 1800;            // 30min TTL on finished entries
pub const WATCH_MAX_PER_WINDOW: u32 = 8;               // rate-limit cap
pub const WATCH_WINDOW_SECONDS: u64 = 10;              // rolling window
pub const WATCH_OVERLOAD_KILL_SECONDS: u64 = 45;       // sustained-cap auto-disable

/// A tracked background process.

    pub id: String,                   // "proc_" + 12-hex

    pub command: String,
    pub pid: Option<u32>,

    pub window_hits: u32,
```

**Decisions the planner must make:**
- Watch-notification delivery mechanism: `broadcast` vs `mpsc` vs direct invocation of an agent-loop hook. `broadcast` keeps ProcessRegistry agnostic of the agent-loop; a subscriber task drains into the outbound-turn trigger. Recommend `broadcast` — matches the `tokio::sync::watch` pattern used by `StatusLineState`.

```rust
//! Shared parent/child iteration-budget handle (PROV-10, D-10/D-15/D-16/D-17).
//! SeqCst ordering everywhere (AI-SPEC Pitfall 9; static-grep gate enforces).

use std::sync::Arc;
use std::sync::atomic::{AtomicUsize, Ordering};

#[derive(Clone)]
pub struct BudgetHandle {
    remaining: Arc<AtomicUsize>,
    max: usize,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum PressureTier { None, Caution70, Warning90, Stop100 }

impl BudgetHandle {
    pub fn new(max: usize) -> Self { Self { remaining: Arc::new(AtomicUsize::new(max)), max } }
    pub fn max(&self) -> usize { self.max }
    pub fn remaining(&self) -> usize { self.remaining.load(Ordering::SeqCst) }
    pub fn used(&self) -> usize { self.max.saturating_sub(self.remaining()) }

    /// Decrement. Returns Some(new_remaining) on success; None when already 0 (D-15 Stop tier).
    pub fn consume(&self) -> Option<usize> {
        let prev = self.remaining.fetch_sub(1, Ordering::SeqCst);
        if prev == 0 { self.remaining.fetch_add(1, Ordering::SeqCst); None }
        else { Some(prev - 1) }
    }

    pub fn pressure(&self) -> PressureTier {
        if self.max == 0 { return PressureTier::None; }
        let used_pct = (self.used() * 100) / self.max;
        match used_pct {
            p if p >= 100 => PressureTier::Stop100,
            p if p >= 90  => PressureTier::Warning90,
            p if p >= 70  => PressureTier::Caution70,
            _              => PressureTier::None,
        }
    }
}

pub const CAUTION_ADVISORY: &str = "You have used approximately 70% of your iteration budget. Consider consolidating remaining work and moving toward a final answer.";
pub const WARNING_ADVISORY: &str = "You have used approximately 90% of your iteration budget. Respond with your final answer now.";
```

### Coverage sampling frequency (Nyquist sense)

Each validation target is either a **discrete-event** assertion (fires once; assert on occurrence) or a **continuous/sampled** assertion (sample over a duration; assert aggregate).

| Transcript last-line == `cancelled` event on cancel path | Discrete | Assert once after cancel + 100ms drain. |
| `BudgetHandle` sum-of-consumes equals max | Discrete | One-shot after all join handles complete (stress = 10k loops; sampling rate = once at end). |

### 3. Watch-pattern notification delivery: broadcast channel vs. direct agent-loop hook?

The CLI's agent-loop outer driver (`main.rs:run_chat`) spawns a subscriber task that drains the broadcast and bumps the `tui.update_status_line(...)` / appends a pseudo-tool-result to the next turn's user message. Exact insertion point: planner defers to the `hermes-agent` reference behaviour but the channel seam is clean. **Flag for user confirmation if a specific agent-loop hook already exists (the `ChildToolProgressCallback` pattern could be reused for watch events too — but that's cross-wiring tools and exec; broadcast is cleaner).**
