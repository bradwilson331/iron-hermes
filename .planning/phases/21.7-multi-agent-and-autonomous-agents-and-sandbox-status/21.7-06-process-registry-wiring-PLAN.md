     ```rust
     /// Feed a chunk of child output (already decoded UTF-8-safe or replaced)
     /// into the process's rolling buffer; scan for watch-pattern matches,
     /// consult the rate limiter, and broadcast matched events on Fire.
     /// Called by the stdout/stderr drain tasks spawned in Plan 06 (terminal + execute_code).
     pub fn ingest_output(&mut self, id: &str, chunk: &str) {
         let patterns_snapshot: Vec<Regex> = {
             match self.running.get(id) {
                 Some(s) => s.watch_patterns.clone(),
                 None => return,
             }
         };

         // Split chunk by lines for regex matching; append to buffer regardless.
         if let Some(s) = self.running.get_mut(id) {
             append_bounded_utf8(&mut s.output_buffer, chunk, MAX_OUTPUT_CHARS);
         }

         if patterns_snapshot.is_empty() { return; }

         for line in chunk.lines() {
             for (idx, pat) in patterns_snapshot.iter().enumerate() {
                 if pat.is_match(line) {
                     let decision = self.rate_limit_check(id);
                     if matches!(decision, RateDecision::Fire) {
                         let pid_opt = self.running.get(id).and_then(|s| s.pid);
                         if let Some(pid) = pid_opt {
                             let _ = self.watch_tx.send(WatchEvent {
                                 session_id: id.to_string(),
                                 pid,
                                 matched_line: line.to_string(),
                                 pattern_index: idx,
                             });
                         }
                     }
                     break; // one match per line is enough
                 }
             }
         }
     }
     ```

   //! E-03 / S-01 / S-02 — end-to-end: spawn children, drain, assert no-zombies.

   use ironhermes_exec::process_registry::*;
   use std::sync::Arc;
   use std::time::Duration;
   use tokio::sync::RwLock;

   #[tokio::test]
   #[cfg(unix)]
   async fn drain_kills_all_tracked_children_no_zombies() {
       use sysinfo::{System, SystemExt, ProcessExt, Pid};
       use nix::sys::wait::{waitpid, WaitStatus, WaitPidFlag};
       use nix::unistd::Pid as NixPid;

       let reg = Arc::new(RwLock::new(ProcessRegistry::new_for_session("t-integ")));

       // Spawn 3 "sleep 60" children.
       let mut pids = Vec::new();
       for _ in 0..3 {
           let id = {
               let mut r = reg.write().await;
               r.spawn(SpawnSpec {
                   command: "sleep 60".into(),
                   cwd: None, env: vec![], watch_patterns: vec![],
               }).await.expect("spawn")
           };
           let pid_opt = {
               let r = reg.read().await;
               r.snapshot().entries.iter().find(|e| e.id == id).and_then(|e| e.pid)
           };
           pids.push(pid_opt.expect("pid"));
       }

       // Sanity: sysinfo sees them alive.
       let mut sys = System::new_all();
       sys.refresh_all();
       for pid in &pids {
           assert!(sys.process(Pid::from(*pid as usize)).is_some(),
               "pre-drain sanity: pid {} should be alive", pid);
       }

       // Drain.
       reg.write().await.drain_and_kill_session("t-integ").await.expect("drain");

       // Give the kill→wait→reap a moment to settle.
       tokio::time::sleep(Duration::from_millis(500)).await;

       // Post-drain: sysinfo sees none of them.
       sys.refresh_all();
       for pid in &pids {
           let alive = sys.process(Pid::from(*pid as usize));
           assert!(alive.is_none(),
               "E-03 / S-01: pid {} must be gone after drain_and_kill_session. found={:?}",
               pid, alive.map(|p| p.status()));
       }

       // No zombies (WNOHANG on a zombied child returns Ok(Exited), on a missing child returns Err ECHILD).
       for pid in &pids {
