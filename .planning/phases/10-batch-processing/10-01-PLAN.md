        BatchCommands::Run { input, output, workers, model } => {

        BatchCommands::Status => {
            runner::cmd_status().await
        }
        BatchCommands::Cancel => {
            runner::cmd_cancel().await
        }
        BatchCommands::List => {
            runner::cmd_list().await

    }

**1. Implement checkpoint.rs** (D-05, D-06):

```rust
use anyhow::{Context, Result};
use sha2::{Digest, Sha256};
use std::collections::HashMap;
use std::path::Path;
use super::types::CheckpointEntry;

/// SHA-256 hash of the prompt string, returned as 64-char lowercase hex (D-05).
pub fn prompt_hash(prompt: &str) -> String {
    let mut hasher = Sha256::new();
    hasher.update(prompt.as_bytes());
    format!("{:x}", hasher.finalize())
}

/// Load checkpoint from JSON file. Returns empty map if file doesn't exist (D-06).
pub fn load_checkpoint(path: &Path) -> Result<HashMap<String, CheckpointEntry>> {
    if !path.exists() {
        return Ok(HashMap::new());
    }
    let data = std::fs::read_to_string(path)
        .with_context(|| format!("Failed to read checkpoint: {}", path.display()))?;
    if data.trim().is_empty() {
        return Ok(HashMap::new());
    }
    serde_json::from_str(&data)
        .with_context(|| format!("Failed to parse checkpoint: {}", path.display()))
}

/// Atomically save checkpoint to JSON file (temp + rename to prevent corruption).
pub fn save_checkpoint(path: &Path, data: &HashMap<String, CheckpointEntry>) -> Result<()> {
    let tmp = path.with_extension("checkpoint.tmp");
    let json = serde_json::to_string_pretty(data)?;
    std::fs::write(&tmp, &json)?;
    std::fs::rename(&tmp, path)?;
    Ok(())
}
```

**2. Implement sharegpt.rs** (D-07, D-08):

```rust
use ironhermes_core::{ChatMessage, Role};
use super::types::ShareGptTurn;

/// Convert AgentResult messages to ShareGPT conversation turns (D-07, D-08).
/// Roles: User -> "human", Assistant text -> "gpt", tool_calls -> "tool_call",
/// Tool -> "tool_response". System messages are skipped.
pub fn messages_to_sharegpt(messages: &[ChatMessage]) -> Vec<ShareGptTurn> {
    let mut turns = Vec::new();
    for msg in messages {
        match msg.role {
            Role::User => {
                if let Some(text) = msg.content_text() {
                    turns.push(ShareGptTurn {
                        from: "human".to_string(),
                        value: text.to_string(),
                    });
                }
            }
            Role::Assistant => {
                // Text content as "gpt" turn
                if let Some(text) = msg.content_text() {
                    if !text.is_empty() {
                        turns.push(ShareGptTurn {
                            from: "gpt".to_string(),
                            value: text.to_string(),
                        });
                    }
                }
                // Tool calls as separate "tool_call" turns (D-07)
                if let Some(tool_calls) = &msg.tool_calls {
                    for tc in tool_calls {
                        turns.push(ShareGptTurn {
                            from: "tool_call".to_string(),
                            value: serde_json::to_string(tc).unwrap_or_default(),
                        });
                    }
                }
            }
            Role::Tool => {
                turns.push(ShareGptTurn {
                    from: "tool_response".to_string(),
                    value: msg.content_text().unwrap_or("").to_string(),
                });
            }
            Role::System => {} // System prompt excluded from ShareGPT trajectory
        }
    }
    turns
}
```

**3. Implement runner.rs** (D-01 `run`, D-02 `status`, D-03 `cancel`, D-04 `list`, D-05/D-06 checkpoint, D-09/D-10 JSONL output):

The runner module implements:

- `cmd_run(input, output, workers, model)`: Main batch execution.
  1. Load config via `Config::load()`. Resolve output path (default: `batch_output/<input_stem>_output.jsonl`). Create output dir if needed.
  2. Compute checkpoint path as `{output_path}.checkpoint.json`. Call `load_checkpoint()`.
  3. Open input JSONL with `tokio::io::BufReader`. Read line by line. For each line: parse as `BatchEntry`, compute `prompt_hash(&entry.prompt)`, skip if hash present in checkpoint.
  4. Create `Arc<Semaphore>` with `workers` count (CLI flag > config > default 4).
  5. Create `mpsc::channel::<(TrajectoryLine, String)>(256)` — tuple of trajectory + content hash.
  6. Spawn a single writer task that receives from the channel. For each received trajectory:
     - If `trajectory.quality.passed`: append JSON line to output file.
     - Else: append to reject file `{output_path%.jsonl}_rejected.jsonl` (D-11).
     - Update checkpoint HashMap with hash -> `CheckpointEntry { status, timestamp }`.
     - Call `save_checkpoint()` after every entry (D-05).
  7. Create a `JoinSet`. For each non-skipped entry:
     - `semaphore.clone().acquire_owned().await` to get permit.
     - Spawn task: build `messages` vec with `ChatMessage::user(entry.prompt)`, optionally prepend system message. Create `LlmClient` from config. Create `AgentLoop::new(client, registry.clone(), max_turns)`. Call `loop.run(messages).await`.
     - Convert `AgentResult` to `TrajectoryLine`: call `messages_to_sharegpt(&result.messages)`, populate id (content hash), model, timestamp (`chrono::Utc::now().to_rfc3339()`), usage from `result.total_usage`, turns from `result.turns_used`.
     - For now, set `quality: QualityResult { passed: true, reasons: vec![] }` — Plan 02 adds real filtering.
     - Send trajectory + hash through channel.
     - Drop permit (automatic when task ends).
  8. Drop sender. `join_set.join_all().await`. `writer_handle.await`.
