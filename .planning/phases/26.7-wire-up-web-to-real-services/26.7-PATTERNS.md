```rust
pub async fn get_memory() -> Result<MemoryInfo> {
    let state = crate::server::state::global_app_state();
    let Some(ref mgr) = state.memory_manager else {
        return Ok(MemoryInfo::default());
    };
    let entries = mgr.lock().await.to_memory_entries().await;
    // entries is MemoryEntries { entries: HashMap<MemoryTarget, Vec<String>> }

        agent_entries: map_entries(&entries, ironhermes_core::memory_store::MemoryTarget::Memory),
        user_entries:  map_entries(&entries, ironhermes_core::memory_store::MemoryTarget::User),
    })
}
```

**Source:** `sessions.rs` lines 33, 48-51

```rust
let resource = use_server_future(crate::server::api::fn_name)?;
```

The `?` on `use_server_future(...)` suspends the component until at least one result arrives (Dioxus 0.7 suspension pattern).
