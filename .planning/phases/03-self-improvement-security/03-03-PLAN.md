       ```rust
       //! SSRF (Server-Side Request Forgery) URL validation.
       //! Port of hermes-agent/tools/url_safety.py.
       //! DNS rebinding is a known limitation (TOCTOU between resolution and connection) — D-17.

       use std::net::{IpAddr, Ipv4Addr, ToSocketAddrs};
       use std::sync::LazyLock;
       use std::collections::HashSet;
       use tracing::warn;
       use url::Url;
       ```

       **Constants**:
       ```rust
       static BLOCKED_HOSTNAMES: LazyLock<HashSet<&'static str>> = LazyLock::new(|| {
           ["metadata.google.internal", "metadata.goog"].into_iter().collect()
       });

       // CGNAT range 100.64.0.0/10 — not covered by IpAddr::is_private()
       const CGNAT_START: u32 = 0x6440_0000; // 100.64.0.0
       const CGNAT_END: u32   = 0x647F_FFFF; // 100.127.255.255
       ```

       **`pub fn is_safe_url(url_str: &str) -> bool`**:
       - Parse URL with `url::Url::parse(url_str)` — return `false` on parse error (fail closed)
       - Extract hostname — return `false` if no host
       - Check hostname against `BLOCKED_HOSTNAMES` — return `false` if match (D-18)
       - Resolve hostname via DNS: `(hostname, 0u16).to_socket_addrs()` — return `false` on resolution error (fail closed, D-16)
       - Check EVERY resolved IP against `is_blocked_ip()` — return `false` if ANY is blocked
       - Return `true` only if all IPs are safe
       - Log blocked URLs via `warn!("SSRF blocked: {} resolved to blocked IP", url_str)`

       **`fn is_blocked_ip(ip: IpAddr) -> bool`**:
       ```rust
       match ip {
           IpAddr::V4(v4) => {
               v4.is_private()
                   || v4.is_loopback()
                   || v4.is_link_local()
                   || v4.is_broadcast()
                   || v4.is_multicast()
                   || v4.is_unspecified()
                   || is_cgnat(v4)
           }
           IpAddr::V6(v6) => {
               v6.is_loopback() || v6.is_multicast() || v6.is_unspecified()
           }
       }
       ```

       **`fn is_cgnat(ip: Ipv4Addr) -> bool`**:
       ```rust
       let bits: u32 = ip.into();
       bits >= CGNAT_START && bits <= CGNAT_END
       ```

       **IMPORTANT**: `to_socket_addrs()` is synchronous. Add a doc comment noting that callers in async contexts should use `tokio::task::spawn_blocking(|| is_safe_url(url))` or switch to `tokio::net::lookup_host`. The function itself remains sync because Phase 4 will handle the async wrapping at the call site.

    2. **Create `crates/ironhermes-gateway/src/rate_limiter.rs`**:

       ```rust
       use std::collections::HashMap;
       use std::sync::{Arc, Mutex};
       use std::time::Instant;

       #[derive(Clone)]
       struct TokenBucketState {
           tokens: f64,
           last_refill: Instant,
       }

       /// Per-user inbound rate limiter using token bucket algorithm (D-20).
       /// Excess messages are silently dropped (D-21).
       #[derive(Clone)]
       pub struct PerUserRateLimiter {
           state: Arc<Mutex<HashMap<String, TokenBucketState>>>,
           messages_per_minute: f64,
           burst_size: f64,
       }
       ```

       **`PerUserRateLimiter::new(messages_per_minute: u32, burst_size: u32) -> Self`**: Stores config, creates empty HashMap.

       **`PerUserRateLimiter::check_and_consume(&self, user_id: &str) -> bool`**:
       - Lock the state HashMap
       - Get or insert `TokenBucketState` for this `user_id` (initial tokens = burst_size)
       - Compute elapsed seconds since `last_refill`
       - Refill tokens: `tokens = min(tokens + elapsed * messages_per_minute / 60.0, burst_size)`
       - Update `last_refill` to now
       - If `tokens >= 1.0`: consume one token, return `true`
       - Else: return `false` (silently drop, D-21)
