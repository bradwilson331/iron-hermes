```rust
}
```

```rust
pub fn format_entries_for_prompt(entries: &MemoryEntries, target: MemoryTarget) -> Option<String> {
    let entry_list = entries.entries.get(&target)?;
    if entry_list.is_empty() {
        return None;
    }
    let header = match target {
        MemoryTarget::Memory => "## Memory",
        MemoryTarget::User => "## User Profile",
    };
    Some(format!("{}\n\n{}", header, entry_list.join("\n")))
}
```

```rust
use crate::constants::{get_hermes_home, MEMORY_CHAR_LIMIT, USER_CHAR_LIMIT};
use crate::memory_store::MemoryStore;

const MEMORIES_DIR: &str = "memories";

pub fn build_memory_provider(config: &MemoryConfig) -> anyhow::Result<Box<dyn MemoryProvider + Send>> {
    match config.provider.as_str() {
        "file" => {
            let memory_dir = get_hermes_home().join(MEMORIES_DIR);
            Ok(Box::new(MemoryStore::new(memory_dir)))
        }
        "sqlite" | "grafeo" | "duckdb" => {
            anyhow::bail!(
                "Memory provider '{}' is not available in this build. \
                 It requires the '{}' feature flag. Available providers: file.",

            )

        other => {
            anyhow::bail!(

                 Future providers (sqlite, grafeo, duckdb) require feature flags.",

            )
        }
    }
}
```

```rust
use crate::memory_store::MemoryStore;

#[async_trait]
impl MemoryProvider for MemoryStore {
    async fn initialize(&mut self, _config: &MemoryProviderConfig) -> Result<()> {
        self.load_from_disk()?;
        Ok(())
    }

    async fn prefetch(&self, _session_id: &str) -> Result<MemoryEntries> {
        // File provider: entries are already in memory after load_from_disk.
        // We need access to the internal entries. Add a pub(crate) getter.
        Ok(self.to_memory_entries())
    }

    async fn sync_turn(&self, _session_id: &str, _entries: &MemoryEntries) -> Result<()> {
        Ok(())  // File provider: disk is authoritative, mutations write directly
    }

    async fn on_session_end(&self, _session_id: &str, _entries: &MemoryEntries) -> Result<()> {
        Ok(())  // File provider: writes happen in-place via add/replace/remove
    }

    async fn shutdown(&mut self) -> Result<()> {
        Ok(())  // File provider: no resources to release
    }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| config.yaml -> provider factory | User-supplied provider string parsed at startup |
| MemoryEntries data flow | Entries pass through trait hooks; file provider inherits existing security scanning |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-01 | Tampering | config.yaml memory.provider | mitigate | Exhaustive match in build_memory_provider rejects unknown provider strings with hard error (D-09); no arbitrary code execution path |
| T-11-02 | Information Disclosure | MemoryProviderConfig.extra | accept | Extra settings are HashMap<String, Value>; no secrets stored here — API keys handled by provider resolution (Phase 12) |
| T-11-03 | Spoofing | MemoryProvider trait | accept | Trait is internal to the binary; providers are compiled-in, not dynamically loaded; no plugin attack surface |
| T-11-04 | Tampering | MemoryEntries passed to hooks | accept | Entries are already security-scanned by MemoryStore::add/replace before reaching sync_turn/on_session_end; scanning remains in the mutation path, not the trait hooks |
</threat_model>
