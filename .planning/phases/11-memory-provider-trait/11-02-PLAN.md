```rust
use async_trait::async_trait;
use std::collections::HashMap;
use std::path::PathBuf;
use anyhow::Result;
use crate::memory_store::{MemoryTarget, MemoryResult};

#[derive(Debug, Clone, Default)]
pub struct MemoryEntries {
    pub entries: HashMap<MemoryTarget, Vec<String>>,
}

#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
pub struct MemoryProviderConfig {
    pub provider: String,
    pub memory_dir: PathBuf,
    pub memory_char_limit: usize,
    pub user_char_limit: usize,
    #[serde(default)]
    pub extra: HashMap<String, serde_json::Value>,
}

#[async_trait]
pub trait MemoryProvider: Send + Sync + 'static {
    // Lifecycle hooks (async)
    async fn initialize(&mut self, config: &MemoryProviderConfig) -> Result<()>;
    async fn prefetch(&self, session_id: &str) -> Result<MemoryEntries>;
    async fn sync_turn(&self, session_id: &str, entries: &MemoryEntries) -> Result<()>;
    async fn on_session_end(&self, session_id: &str, entries: &MemoryEntries) -> Result<()>;
    async fn shutdown(&mut self) -> Result<()>;

    // Operational methods (sync)
    fn load_from_disk(&mut self) -> Result<()>;
    fn add(&mut self, target: MemoryTarget, content: &str) -> MemoryResult;
    fn replace(&mut self, target: MemoryTarget, old_text: &str, new_content: &str) -> MemoryResult;
    fn remove(&mut self, target: MemoryTarget, old_text: &str) -> MemoryResult;
    fn format_for_system_prompt(&self, target: MemoryTarget) -> Option<String>;
    fn to_memory_entries(&self) -> MemoryEntries;
}

pub fn format_entries_for_prompt(entries: &MemoryEntries, target: MemoryTarget) -> Option<String>;
pub fn build_memory_provider(config: &MemoryConfig) -> anyhow::Result<Box<dyn MemoryProvider + Send>>;
```

Wait — `build_memory_provider` returns `Box<dyn MemoryProvider + Send>`. To put it in `Arc<Mutex<>>`, we need to unwrap the Box. Use: `Arc::new(Mutex::new(*provider))` — but you can't move out of a `Box<dyn Trait>` into `Mutex::new()` because `dyn Trait` is unsized. Instead, the cleanest approach is to change the pattern:

Option A: Store `Box<dyn MemoryProvider + Send>` inside the Mutex:
```rust
let memory_store: Arc<Mutex<Box<dyn MemoryProvider + Send>>> = Arc::new(Mutex::new(provider));
```
This changes the type everywhere from `Arc<Mutex<dyn MemoryProvider + Send>>` to `Arc<Mutex<Box<dyn MemoryProvider + Send>>>`. Slightly more verbose but works.

Option B: Have build_memory_provider return the concrete type and let caller wrap:
Keep `build_memory_provider` returning `Box<dyn MemoryProvider + Send>`. At the call site:
```rust
let memory_store: Arc<Mutex<dyn MemoryProvider + Send>> = Arc::from(Mutex::new(provider));
```
Actually `Arc::from(Mutex::new(Box))` doesn't give us what we want.

Option C (PREFERRED): For the file provider, construct directly. Validate config with the factory pattern, then construct:
```rust

    }

}

}

```

ACTUALLY the simplest correct approach: keep `build_memory_provider` as-is (returns `Box<dyn MemoryProvider + Send>`), and use `Arc<Mutex<Box<dyn MemoryProvider + Send>>>` as the shared type. But this changes the type signature everywhere which is verbose.

Use the factory for config VALIDATION (D-09), then construct the concrete type directly and coerce at the Arc::new call site. This is valid because only "file" provider exists.
