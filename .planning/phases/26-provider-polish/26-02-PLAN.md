    ```rust
    use std::sync::OnceLock;
    use std::collections::HashSet;

    /// Phase 26 D-12: emit a one-shot per-process stderr banner for legacy env var fallback.
    /// Inserts provider_name into the WARNED set on first call; subsequent calls for the
    /// same provider are silent. Per Resolution #2, the once-only assertion lives in
    /// the Plan 04 subprocess test (OnceLock can't be reset between unit tests in the
    /// same process — banner-once-only is verified across two `hermes status` invocations).
    fn legacy_env_with_banner(provider_name: &str, var_name: &str) -> Option<String> {
        static WARNED: OnceLock<std::sync::Mutex<HashSet<String>>> = OnceLock::new();
        let warned = WARNED.get_or_init(|| std::sync::Mutex::new(HashSet::new()));
        let value = std::env::var(var_name).ok()?;
        let mut set = warned.lock().unwrap();
        if set.insert(provider_name.to_string()) {
            eprintln!(
                "[provider:{}] using deprecated env var {} — set providers.{}.api_key_env in config.yaml to silence this warning",
                provider_name, var_name, provider_name
            );
        }
        Some(value)
    }

    /// Phase 26 D-13: one-shot banner for config.model.api_key literal field.
    /// NEVER prints the key value (T-26-01) — only the config path.
    fn warn_model_api_key_deprecated_once() {
        static WARNED: OnceLock<std::sync::atomic::AtomicBool> = OnceLock::new();
        let flag = WARNED.get_or_init(|| std::sync::atomic::AtomicBool::new(false));
        if !flag.swap(true, std::sync::atomic::Ordering::SeqCst) {
            eprintln!(
                "[config:model.api_key] deprecated — set providers.<main-provider>.api_key_env instead. Will be removed in a future major release."
            );
        }
    }

    /// Phase 26 D-01 / Pitfall 5 / Resolution #3: one-shot per-provider banner for
    /// the deprecated inline `providers.NAME.api_key` literal field. NEVER prints the
    /// key value (T-26-01) — only the provider name.
    fn warn_provider_api_key_inline_once(provider_name: &str) {
        static WARNED: OnceLock<std::sync::Mutex<HashSet<String>>> = OnceLock::new();
        let warned = WARNED.get_or_init(|| std::sync::Mutex::new(HashSet::new()));
        let mut set = warned.lock().unwrap();
        if set.insert(provider_name.to_string()) {
            eprintln!(
                "[provider:{}] config.providers.{}.api_key inline literal — prefer api_key_env reference (will be removed in a future major release)",
                provider_name, provider_name
            );
        }
    }
    ```

    ```rust
    // --- 4. Resolve API keys with precedence (Phase 26 D-11 leak fix + D-12 legacy banners) ---
    // Precedence per provider:
    //   1. providers.NAME.api_key_env -> std::env::var(<that name>)              [D-01 — preferred]
    //   2. providers.NAME.api_key (deprecated literal field, banner emitted)     [Resolution #3]
    //   3. For built-ins ONLY: legacy OPENAI_API_KEY / ANTHROPIC_API_KEY /
    //      OPENROUTER_API_KEY env var with one-shot stderr deprecation banner   [D-12]
    //   4. For main provider only: config.model.api_key (deprecated, banner)    [D-13]
    //   5. None — caller fails on first LLM call with "no key configured for X" [D-11]
    let main_name = config.model.provider.clone();

    for (name, endpoint) in endpoints.iter_mut() {
        let provider_cfg = config.providers.get(name);

        // (1) Explicit api_key_env reference (preferred)
        let env_key_via_alias: Option<String> = provider_cfg
            .and_then(|p| p.api_key_env.as_ref())
            .and_then(|env_name| {
                // Validate env var name shape — defense in depth (T-26-04).
                // Validation already ran at the start of build() in Task 2; this is an
                // ASCII safety net before std::env::var() is called.
                if crate::config::validate_api_key_env(env_name).is_err() {
                    return None;
                }
                std::env::var(env_name).ok()
            });

        // (2) Inline literal api_key field (deprecated; banner)
        let inline_literal_key: Option<String> = provider_cfg.and_then(|p| {
            if p.api_key.is_some() {
                warn_provider_api_key_inline_once(name);
            }
            p.api_key.clone()
        });

        // (3) Built-in legacy env-var fallback (deprecated; banner)
        //     D-11: NO wildcard arm. Custom providers fall through to None.
        let legacy_env_key: Option<String> = if env_key_via_alias.is_none() && inline_literal_key.is_none() {
            match name.as_str() {
                "openai" => legacy_env_with_banner(name, "OPENAI_API_KEY"),
                "anthropic" => legacy_env_with_banner(name, "ANTHROPIC_API_KEY"),
                "openrouter" => legacy_env_with_banner(name, "OPENROUTER_API_KEY"),
                _ => None, // D-11: NO `_ => std::env::var("OPENAI_API_KEY")` LEAK
            }
        } else {
            None
        };

        // (4) config.model.api_key — main provider only, deprecated
        let model_api_key: Option<String> = if name == &main_name && config.model.api_key.is_some() {
            warn_model_api_key_deprecated_once();
            config.model.api_key.clone()
        } else {
            None
        };

        endpoint.api_key = env_key_via_alias
            .or(inline_literal_key)
            .or(legacy_env_key)
            .or(model_api_key);
    }
    ```
