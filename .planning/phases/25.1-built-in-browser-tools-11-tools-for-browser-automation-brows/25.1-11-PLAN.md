// and restores on Drop. Use verbatim for OPENAI_API_KEY/ANTHROPIC_API_KEY in test 3.
struct EnvGuard { key: &'static str, prev: Option<String> }
impl EnvGuard {
    fn set(key: &'static str, val: &str) -> Self {
        let prev = std::env::var(key).ok();
        unsafe { std::env::set_var(key, val) };
        Self { key, prev }
    }
}
impl Drop for EnvGuard {
    fn drop(&mut self) {
        unsafe {
            match &self.prev {
                Some(v) => std::env::set_var(self.key, v),
                None => std::env::remove_var(self.key),
            }
        }
    }
}

//! Phase 25.1 D-20: Three mandatory integration tests for the browser toolset.
//!
//! Pattern: in-process ToolRegistry + chromiumoxide + wiremock-hosted local HTML.
//! Skips (not fails) when chromium binary is unavailable (D-22).

use std::sync::OnceLock;

use ironhermes_tools::browser_session::{find_chromium_binary, BrowserSession};
use serde_json::json;
use wiremock::matchers::{any, method, path};
use wiremock::{Mock, MockServer, ResponseTemplate};

fn env_lock() -> &'static std::sync::Mutex<()> {
    static LOCK: OnceLock<std::sync::Mutex<()>> = OnceLock::new();
    LOCK.get_or_init(|| std::sync::Mutex::new(()))
}

/// RAII env var guard — restores original on drop, even on panic.
/// Verbatim from crates/ironhermes-cli/tests/provider_integration.rs lines 24-47.
struct EnvGuard {
    key: &'static str,
    prev: Option<String>,
}

impl EnvGuard {
    fn set(key: &'static str, val: &str) -> Self {
        let prev = std::env::var(key).ok();
        // SAFETY: test-only env mutation, serialised behind env_lock().
        unsafe { std::env::set_var(key, val) };
        Self { key, prev }
    }
}

impl Drop for EnvGuard {
    fn drop(&mut self) {
        unsafe {
            match &self.prev {
                Some(v) => std::env::set_var(self.key, v),
                None => std::env::remove_var(self.key),
            }
        }
    }
}

/// D-22: SKIP guard. Tests return early without failing when chromium is absent.
fn chromium_available() -> bool {
    // The IRONHERMES_BROWSER_TEST_DISABLE escape hatch (added in plan 10) makes
    // find_chromium_binary return None; clear it here so this guard reflects
    // the real machine state.
    if std::env::var("IRONHERMES_BROWSER_TEST_DISABLE").is_ok() {
        return false;
    }
    find_chromium_binary(None).is_some()
}

/// Helper: register the 11 browser tools onto a fresh ToolRegistry.
fn make_browser_registry(
    resolver: std::sync::Arc<ironhermes_core::provider::ProviderResolver>,
) -> (
    ironhermes_tools::ToolRegistry,
    std::sync::Arc<tokio::sync::Mutex<Option<BrowserSession>>>,
) {
    let session = std::sync::Arc::new(tokio::sync::Mutex::new(None::<BrowserSession>));
    let mut registry = ironhermes_tools::ToolRegistry::new();
    registry.register_browser_tools(session.clone(), resolver);
    let mut cfg = ironhermes_core::config::ToolsConfig::default();
    cfg.toolsets.insert(
        "browser".to_string(),
        ironhermes_core::config::ToolsetEntry { enabled: true },
    );
    registry.set_toolset_config(Some(cfg));
    (registry, session)
}

    registry.handle_tool_call(name, args).await
}

#[tokio::test(flavor = "multi_thread")]
async fn browser_navigate_then_snapshot_returns_refs() {
    let _g = env_lock().lock().unwrap_or_else(|p| p.into_inner());
    if !chromium_available() {
        eprintln!("SKIP browser_navigate_then_snapshot_returns_refs: no chromium binary (D-22)");
        return;
    }

    let server = MockServer::start().await;
    Mock::given(method("GET"))
        .and(path("/test-page"))
        .respond_with(ResponseTemplate::new(200).set_body_string(
            r#"<!doctype html><html><body>
                <h1>Test</h1>
                <button>Submit</button>
                <input type="text" placeholder="Email">
            </body></html>"#,
        ))
        .mount(&server)
        .await;

    // Build a default resolver — test 1 doesn't need vision routing.
    let mut config = ironhermes_core::Config::default();
    config.model.provider = "openai".to_string();
    config.model.default = "gpt-4o".to_string();
    let resolver = std::sync::Arc::new(
        ironhermes_core::ProviderResolver::build(&config).expect("resolver build"),
    );
    let (registry, session) = make_browser_registry(resolver);

    let nav_url = format!("{}/test-page", server.uri());
    let nav_result = invoke(&registry, "browser_navigate", json!({"url": nav_url})).await
        .expect("navigate should succeed");
    assert!(nav_result.contains("\"status\":200") || nav_result.contains("\"url\""),
        "navigate response unexpected: {nav_result}");

    let snap = invoke(&registry, "browser_snapshot", json!({})).await
        .expect("snapshot should succeed");

    // D-10 ref-line format: `[N] button "Submit"`
    assert!(snap.contains("button") && snap.contains("Submit"),
        "expected button \"Submit\" in snapshot output, got:\n{snap}");
    let has_ref_line = snap.lines().any(|line| {
        line.trim_start().starts_with('[')
            && line.contains("button")
            && line.contains("Submit")
    });
    assert!(has_ref_line, "expected `[N] button \"Submit\"` ref line, got:\n{snap}");

    // Cleanup so subsequent tests in the same process get a fresh session.
    let _ = invoke(&registry, "browser_close", json!({})).await;
    drop(session);
}

        .expect("snapshot should succeed");

        .find(|l| l.contains("button") && l.contains("Original"))
        .and_then(|l| l.trim_start().strip_prefix('['))

        .and_then(|n| n.parse().ok())
