```rust
//! Phase 25.1 D-04 / D-06 / D-07 / D-08 / D-09: browser_vision.
//!
//! Captures a full-page screenshot via chromiumoxide, encodes to base64 data URL,
//! and routes the multimodal call through `resolve_role("vision")` (Phase 26 D-07
//! cascade) — falling back to the main provider when no vision role is configured.
//!
//! This is the FIRST real consumer of Phase 26's vision-role infrastructure.
//! Closes Phase 26 SC-2 ("vision wired") retroactively.

use std::sync::Arc;

use async_trait::async_trait;
use base64::Engine as _;
use ironhermes_agent::any_client::{build_main_client, build_role_client};
use ironhermes_core::provider::ProviderResolver;
use ironhermes_core::types::{ChatMessage, ContentPart, ImageUrl, MessageContent, Role};
use ironhermes_core::ToolSchema;
use serde_json::json;
use tokio::sync::Mutex;
use tracing::{debug, warn};

use crate::browser_session::{find_chromium_binary, BrowserSession};
use crate::registry::{Prerequisite, Tool};

const DEFAULT_PROMPT: &str =
    "Describe what's visible on this page in detail, including any interactive elements, \
     text content, and visual structure.";

pub struct BrowserVisionTool {
    session: Arc<Mutex<Option<BrowserSession>>>,
    resolver: Arc<ProviderResolver>,
}

impl BrowserVisionTool {
    pub fn new(
        session: Arc<Mutex<Option<BrowserSession>>>,
        resolver: Arc<ProviderResolver>,
    ) -> Self {
        Self { session, resolver }
    }

    /// D-06 helper: vision is available when EITHER a vision role is resolvable OR
    /// the main provider's model metadata declares supports_vision=true.
    fn vision_capable(&self) -> bool {
        if self.resolver.resolve_role("vision").is_some() {
            return true;
        }
        let main = self.resolver.resolve_for_main();
        main.model_metadata
            .as_ref()
            .map(|m| m.supports_vision)
            .unwrap_or(false)
    }
}

#[async_trait]
impl Tool for BrowserVisionTool {
    fn name(&self) -> &str { "browser_vision" }
    fn toolset(&self) -> &str { "browser" }
    fn description(&self) -> &str {
        "Capture a full-page screenshot of the current page and analyze it via the auxiliary \
         vision role (or the main provider if multimodal-capable). Optional `prompt` argument \
         narrows the analysis (e.g. 'What's the price of the highlighted item?'). Default prompt \
         describes the page contents in detail."
    }

    fn schema(&self) -> ToolSchema {
        ToolSchema::new(
            "browser_vision",
            self.description(),
            json!({
                "type": "object",
                "properties": {
                    "prompt": {
                        "type": "string",
                        "description": "Optional analysis prompt. Defaults to a general 'describe this page' query."
                    }
                },
                "required": []
            }),
        )
    }

    fn is_available(&self) -> bool {
        find_chromium_binary(None).is_some() && self.vision_capable()
    }

    fn prerequisites(&self) -> Vec<Prerequisite> {
        vec![
            Prerequisite {
                kind: "binary_present".to_string(),
                name: "chromium-or-chrome".to_string(),
                description: "Chromium or Google Chrome browser binary on PATH or at a standard install location".to_string(),
                required: true,
            },
            Prerequisite {
                kind: "config_field".to_string(),
                name: "auxiliary.vision OR multimodal-capable main provider".to_string(),
                description:
                    "Either set auxiliary.vision: { provider, model } in config.yaml or use a main provider with supports_vision=true (e.g. gpt-4o, claude-3.5-sonnet, gemini-pro)".to_string(),
                required: true,
            },
        ]
    }

    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<String> {
        let prompt = args
            .get("prompt")
            .and_then(|v| v.as_str())
            .unwrap_or(DEFAULT_PROMPT)
            .to_string();

        debug!(prompt_len = prompt.len(), "browser_vision invoked");

        // 1. Capture screenshot via chromiumoxide.
        let mut guard = self.session.lock().await;
        let sess = ensure_session(&mut guard).await?;

        use chromiumoxide::page::ScreenshotParams;
        use chromiumoxide_cdp::cdp::browser_protocol::page::CaptureScreenshotFormat;

        let screenshot_bytes: Vec<u8> = sess
            .page
            .screenshot(
                ScreenshotParams::builder()
                    .format(CaptureScreenshotFormat::Png)
                    .full_page(true)
                    .build(),
            )
            .await
            .map_err(|e| anyhow::anyhow!("screenshot failed: {e}"))?;

        debug!(bytes = screenshot_bytes.len(), "screenshot captured");

        // 2. base64 encode + data URL.
        let b64 = base64::engine::general_purpose::STANDARD.encode(&screenshot_bytes);
        let data_url = format!("data:image/png;base64,{}", b64);

        // 3. Resolve vision client via D-07 cascade. Drop the session lock BEFORE the LLM
        //    call to avoid holding chromium across a multi-second network round-trip.
        drop(guard);

        let client = match build_role_client(&self.resolver, "vision") {
            Ok(Some(c)) => c,
            Ok(None) => build_main_client(&self.resolver)
                .map_err(|e| anyhow::anyhow!("no vision role configured AND main client build failed: {e}"))?,
            Err(e) => {
                warn!(error = %e, "build_role_client('vision') errored; falling back to main");
                build_main_client(&self.resolver)
                    .map_err(|e| anyhow::anyhow!("vision fallback to main failed: {e}"))?
            }
        };

        // 4. Build multimodal ChatMessage. ImageUrl::url is the data: URL — AnthropicClient
        //    converts it to image source.base64 per plan 01 OQ-2 fix.
        let user_msg = ChatMessage {
            role: Role::User,
            content: Some(MessageContent::Parts(vec![
                ContentPart::Text { text: prompt.clone() },
                ContentPart::ImageUrl {
                    image_url: ImageUrl {
                        url: data_url,
                        detail: None,
                    },
                },
            ])),
            tool_calls: None,
            tool_call_id: None,
            name: None,
        };

        // 5. Non-streaming chat completion. tools=None, max_tokens=4096 keeps cost bounded
        //    (a vision describe tends to fit easily in 1500-3000 tokens).
        let response = client
            .chat_completion(&[user_msg], None, None, Some(4096), None, None)
            .await
            .map_err(|e| anyhow::anyhow!("vision LLM call failed: {e}"))?;

        // 6. Extract response text.
        let text = response
            .choices
            .first()
            .and_then(|c| c.message.content.as_ref())
            .and_then(|c| c.as_text())
            .map(|s| s.to_string())
            .unwrap_or_else(|| "(no response text)".to_string());

        Ok(json!({
            "prompt": prompt,
            "screenshot_bytes": screenshot_bytes.len(),
            "analysis": text
        }).to_string())
    }
}

async fn ensure_session<'a>(
    guard: &'a mut tokio::sync::MutexGuard<'_, Option<BrowserSession>>,
) -> anyhow::Result<&'a mut BrowserSession> {
    if guard.is_none() {
        let cfg = ironhermes_core::config::Config::load().unwrap_or_default().browser;
        let new = BrowserSession::spawn(&cfg).await?;
        **guard = Some(new);
    }
    Ok(guard.as_mut().expect("just inserted"))
}
```
