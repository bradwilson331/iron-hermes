    fn handle_toggle(app: &mut App, name: &str, arg: &str) -> SlashOutcome {
        use std::sync::atomic::Ordering;
        let toggle_atomic = |field: &Arc<AtomicBool>, label: &str| -> SlashOutcome {
            let old = field.fetch_xor(true, Ordering::SeqCst);
            let new_state = !old;
            SlashOutcome::Handled(format!("{label}: {}", if new_state { "on" } else { "off" }))
        };
        match name {
            "yolo" => toggle_atomic(&app.yolo_enabled, "YOLO mode"),
            "verbose" => toggle_atomic(&app.verbose_enabled, "Verbose mode"),
            "statusbar" => toggle_atomic(&app.statusbar_enabled, "Status bar"),
            "debug" => toggle_atomic(&app.debug_enabled, "Debug mode"),
            "skin" => {
                if arg.is_empty() {
                    let current = app.skin.read().map(|s| s.clone()).unwrap_or_else(|p| p.into_inner().clone());
                    SlashOutcome::Handled(format!("Current skin: {current}. Usage: /skin <name>"))
                } else {
