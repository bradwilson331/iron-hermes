    fn cmd_rollback(args: &[&str], ctx: &CommandContext) -> CommandResult {
        let history = match &ctx.history {
            Some(h) => h.clone(),
            None => return CommandResult::Output("History not available.".to_string()),
        };
        let n: usize = args.first()
            .and_then(|s| s.parse().ok())
            .unwrap_or(1);
        if n == 0 {
            return CommandResult::Error("Usage: /rollback [N] where N >= 1".to_string());
        }
        let guard = match history.read() { Ok(g) => g, Err(p) => p.into_inner() };
        let pairs_available = guard.len() / 2;
        if n > pairs_available {
            return CommandResult::Error(format!(
                "Cannot rollback {n} pairs — only {pairs_available} pairs available."
            ));
        }
        CommandResult::Output(format!(
            "Will rollback {n} (user, assistant) pair(s): {} messages remaining (post-router hook will apply truncation).",
            guard.len().saturating_sub(n * 2)
        ))
    }
