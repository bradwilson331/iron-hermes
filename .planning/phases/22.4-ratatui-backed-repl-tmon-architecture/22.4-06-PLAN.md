    //! Pure frame-render function for the tui_rata REPL (Phase 22.4).
    //!
    //! Template: /Users/twilson/code/tmon/src/main.rs `ui()` fn (lines 564–624).
    //! IronHermes adaptation uses a 4-chunk vertical layout per CONTEXT
    //! §specifics:
    //! - Min(5) transcript
    //! - Length(1) knight-rider row (rendered only when in-flight)
    //! - Length(1) status pills row (D-10)
    //! - Length(3) tui-textarea input (D-05)
    //!
    //! The function takes `&App` (not `&mut`) so plan 22.4-10's TestBackend
    //! snapshot tests can render deterministically.

    use ansi_to_tui::IntoText;
    use ratatui::{
        layout::{Constraint, Direction, Layout, Position, Rect},
        widgets::{Block, Borders, Paragraph, Wrap},
        Frame,
    };

    use crate::tui_rata::app::App;
    use crate::tui_rata::knight_rider;
    use crate::tui_rata::status_line::render_status_line_ratatui;

    pub fn ui(frame: &mut Frame, app: &App) {
        let chunks = Layout::default()
            .direction(Direction::Vertical)
            .constraints([
                Constraint::Min(5),    // transcript
                Constraint::Length(1), // knight-rider (blank when idle)
                Constraint::Length(1), // status pills
                Constraint::Length(3), // tui-textarea input
            ])
            .split(frame.area());

        render_transcript(frame, app, chunks[0]);
        render_knight_rider(frame, app, chunks[1]);
        render_status(frame, app, chunks[2]);
        render_input(frame, app, chunks[3]);
        render_cursor(frame, app, chunks[3]);
    }

    fn render_transcript(frame: &mut Frame, app: &App, area: Rect) {
        let title = format!("Chat [{}]", app.scroll_indicator(area));
        let block = Block::default().borders(Borders::ALL).title(title);
        let text = app.transcript_text();
        let paragraph = Paragraph::new(text)
            .block(block)
            .wrap(Wrap { trim: false })
            .scroll((app.transcript_scroll, 0));
        frame.render_widget(paragraph, area);
    }

    fn render_knight_rider(frame: &mut Frame, app: &App, area: Rect) {
        if app.pending_rx.is_none() {
            frame.render_widget(Paragraph::new(""), area);
            return;
        }
        let ansi_string = knight_rider::frame(app.knight_rider_tick);
        let text = ansi_string.as_bytes().into_text().unwrap_or_default();
        frame.render_widget(Paragraph::new(text), area);
    }

    fn render_status(frame: &mut Frame, app: &App, area: Rect) {
        let line = render_status_line_ratatui(&app.status);
        frame.render_widget(Paragraph::new(line), area);
    }

    fn render_input(frame: &mut Frame, app: &App, area: Rect) {
        frame.render_widget(&app.textarea, area);
    }

    fn render_cursor(frame: &mut Frame, app: &App, area: Rect) {
        // tui-textarea exposes `cursor() -> (row, col)` (RESEARCH §Cursor position).
        let (row, col) = app.textarea.cursor();
        // +1 offsets account for the tui-textarea block border (default block).
        // If app.textarea was constructed WITHOUT a block, change offsets to 0 and
        // re-snapshot plan 22.4-10. Plan 22.4-05 Task 1 does NOT set a block on
        // the textarea — but RESEARCH §Cursor position example uses +1 for a
        // bordered block. Confirm with snapshot test.
        let cursor_x = area.x.saturating_add(col as u16);
        let cursor_y = area.y.saturating_add(row as u16);
        frame.set_cursor_position(Position::new(cursor_x, cursor_y));
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| `knight_rider::frame` ANSI bytes → `ansi_to_tui::IntoText` → ratatui Paragraph | Non-SGR ANSI escapes in tool-output lookalikes could target the terminal; `ansi-to-tui` strips non-SGR per RESEARCH §Threat 2 |
| `&App` pure read → `frame.render_widget` + `frame.set_cursor_position` | No mutation; thread-safe by borrow checker |
| App state → transcript render | User-submitted content is plain Line::from(String); ANSI bytes in user input do not get interpreted (RESEARCH §Threat 5) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|

| T-22.4-06-02 | Denial of service | `frame.render_widget(&app.textarea, ...)` panics if textarea holds malformed state | mitigate | tui-textarea is a well-tested widget; App::new always initializes it via `TextArea::default()` → safe. Acceptance: `ui_renders_four_chunks_in_80x24` test proves no panic on empty App. |
