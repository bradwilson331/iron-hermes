use colored::Colorize;
pub const TRACK_WIDTH: usize = 10;

pub fn frame(tick: u64) -> String {
    let period = (TRACK_WIDTH as u64 - 1) * 2;
    let phase = tick % period;
    let lit = if phase < TRACK_WIDTH as u64 { phase as usize } else { (period - phase) as usize };
    (0..TRACK_WIDTH).map(|i| {
        let distance = (i as i32 - lit as i32).unsigned_abs() as usize;
        match distance {
            0 => "█".bright_cyan().to_string(),
            1 => "▓".cyan().to_string(),
            2 => "▒".cyan().dimmed().to_string(),
            _ => "░".dimmed().to_string(),
        }
    }).collect::<String>()
}
```

pub async fn run_chat_ratatui(/* session args */) -> Result<()> {

async fn run_app_inner(terminal: &mut DefaultTerminal, app: &mut App) -> Result<()> {
    let mut events = EventStream::new();   // AFTER ratatui::init() per Pitfall §1 placement note
    let mut tick = time::interval(Duration::from_millis(100));
    tick.set_missed_tick_behavior(time::MissedTickBehavior::Skip);
    let ctrl_c = signal::ctrl_c();
    tokio::pin!(ctrl_c);

    loop {
        let transcript_area = compute_transcript_area(terminal.size()?);
        tokio::select! {
            maybe_event = events.next() => match maybe_event {
                Some(Ok(ev)) => app.handle_event(ev, transcript_area),
                Some(Err(e)) => { tracing::warn!("terminal event error: {e}"); }
                None => { app.should_quit = true; }
            },
            Some(se) = app.pending_rx.as_mut().map_or_else(
                || futures_util::stream::pending().boxed(),
                |rx| rx.recv().into_stream().boxed()
            ) => app.handle_stream_event(se),
            _ = &mut ctrl_c => app.handle_ctrl_c_signal(),
            _ = tick.tick() => app.on_tick(),
        }
        app.reconcile_scroll(transcript_area);  // tmon line 642–643 pattern
        terminal.draw(|f| ui(f, app))?;          // tmon line 644
        if app.should_quit { break; }            // tmon line 645–647
    }
    Ok(())
}

fn submit_turn(app: &mut App) -> Result<()> {
    let (tx, rx) = unbounded_channel::<StreamEvent>();
    app.pending_rx = Some(rx);
    let cancel = app.cancel_parent.child_token();
    app.cancel_child = Some(cancel.clone());
    let tx_delta = tx.clone();
    let agent_loop = app.agent_loop.clone();
    let stream_cb = Box::new(move |delta: &str| { let _ = tx_delta.send(StreamEvent::Delta(delta.to_string())); });
    tokio::spawn(async move {
        let _ = tx.send(StreamEvent::Started);
        let r = agent_loop.run_agent_turn(stream_cb, cancel.clone()).await;
        let _ = tx.send(match r {
            Ok(_)                           => StreamEvent::Finished,
            Err(_) if cancel.is_cancelled() => StreamEvent::Cancelled,
            Err(e)                          => StreamEvent::Error(e.to_string()),
        });
        // tx drops here → channel closes
    });
    Ok(())
}
```
