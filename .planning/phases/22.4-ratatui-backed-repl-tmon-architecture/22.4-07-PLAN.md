## Main loop shape (RESEARCH §tokio::select! canonical)

```rust
let mut events = EventStream::new();
let mut tick = tokio::time::interval(Duration::from_millis(100));
tick.set_missed_tick_behavior(tokio::time::MissedTickBehavior::Skip);
let ctrl_c = tokio::signal::ctrl_c();
tokio::pin!(ctrl_c);

loop {
    let transcript_area = compute_transcript_area(terminal.size()?);
    // Spawn a turn if submit() just stashed pending_rx + pending_tx
    if let (Some(tx), Some(child)) = (app.pending_tx.take(), app.cancel_child.clone()) {
        spawn_turn(&app, tx, child);
    }
    let mut pending_rx = app.pending_rx.as_mut();
    tokio::select! {
        maybe_event = events.next() => match maybe_event {
            Some(Ok(ev)) => app.handle_event(ev, transcript_area),
            Some(Err(e)) => { tracing::warn!("event error: {e}"); }
            None        => { app.should_quit = true; }
        },
        Some(se) = async { pending_rx.as_mut()?.recv().await } => {
            app.handle_stream_event(se);
        },
        _ = &mut ctrl_c => app.handle_ctrl_c_signal(),
        _ = tick.tick() => app.on_tick(),
    }
    app.reconcile_scroll(transcript_area);
    terminal.draw(|f| ui(f, app))?;
    if app.should_quit { break; }
}
```

## Per-turn spawn (RESEARCH §Streaming mpsc Bridge D-17)

```rust
fn spawn_turn(
    app: &App,
    tx: UnboundedSender<StreamEvent>,
    cancel: CancellationToken,
) {
    let agent_loop = app.agent_loop.clone();
    let user_message = app.history.last().cloned(); // or the submitted string — App records it before submit
    let tx_delta = tx.clone();
    let stream_cb: Box<dyn Fn(&str) + Send + Sync> = Box::new(move |delta: &str| {
        let _ = tx_delta.send(StreamEvent::Delta(delta.to_string()));
    });
    tokio::spawn(async move {
        let _ = tx.send(StreamEvent::Started);
        let result = agent_loop.run_agent_turn(
            /* args — exact signature resolved from existing run_chat call site */
            user_message, stream_cb, cancel.clone()
        ).await;
        let terminal_event = match result {
            Ok(_) => StreamEvent::Finished,
            Err(_) if cancel.is_cancelled() => StreamEvent::Cancelled,
            Err(e) => StreamEvent::Error(e.to_string()),
        };
        let _ = tx.send(terminal_event);
        // tx drops → channel closes → main loop observes via terminal event then rx.recv() = None on next iter
    });
}
```
