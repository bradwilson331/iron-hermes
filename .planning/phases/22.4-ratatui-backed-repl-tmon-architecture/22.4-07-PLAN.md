    struct MouseCaptureGuard;
    impl Drop for MouseCaptureGuard {
        fn drop(&mut self) {
            let _ = execute!(io::stdout(), DisableMouseCapture);
        }
    }

    /// Public entry point. Defaults-on for `hermes chat` per D-03.
    ///
    /// D-01 lifecycle sequence:
    /// 1. `install_tui_logger_subscriber` — pre-ratatui (Pitfall 2)
    /// 2. `ratatui::init()` — enables raw mode + EnterAlternateScreen + panic hook
    /// 3. `EnableMouseCapture` + RAII `MouseCaptureGuard` — ratatui::init does NOT enable mouse (RESEARCH Gotcha)
    /// 4. Run main loop (Task 3)
    /// 5. Drop guard → `DisableMouseCapture`; `ratatui::restore()` → `LeaveAlternateScreen` + `disable_raw_mode`
    pub async fn run_chat_ratatui(
        cli: &crate::Cli,
        initial: Option<String>,
        yolo: bool,
    ) -> Result<()> {
        install_tui_logger_subscriber();

        let mut terminal = ratatui::init();

        // D-01 — mouse capture paired with a Drop guard for teardown on all exit paths.
        execute!(io::stdout(), EnableMouseCapture)?;
        let _mouse_guard = MouseCaptureGuard;

        let result = run_with_deps(&mut terminal, cli, initial, yolo).await;

        drop(_mouse_guard);
        ratatui::restore();
        result
    }

    /// Install `tui_logger::TuiTracingSubscriberLayer` as the global tracing
    /// subscriber for the ratatui path. Uses `try_init` so re-install is a
    /// silent no-op (defensive; plan 22.4-08 guarantees no prior install on
    /// the ratatui branch per Pitfall 2).
    fn install_tui_logger_subscriber() {
        use tracing_subscriber::prelude::*;
        let layer = tui_logger::TuiTracingSubscriberLayer;
        let registry = tracing_subscriber::registry().with(layer);
        let _ = registry.try_init();
        let _ = tui_logger::init_logger(log::LevelFilter::Trace);
        tui_logger::set_default_level(log::LevelFilter::Info);
    }

    async fn run_with_deps(
        terminal: &mut DefaultTerminal,
        cli: &crate::Cli,
        initial: Option<String>,
        yolo: bool,
    ) -> Result<()> {
        let deps = build_app_deps(cli, yolo).await?;
        let mut app = App::new(deps);

        if let Some(msg) = initial {
            app.load_history_entry(&msg);
            app.submit();
        }

        run_app_inner(terminal, &mut app).await
    }
