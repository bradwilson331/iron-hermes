// NEW — add to api.rs after create_session()
const HISTORY_LIMIT: usize = 50;

/// Phase 26.7.2 (D-10): Last N messages for the Chat screen history load.
/// Query param style — consistent with all other #[get] routes in this file.
/// Filters to "user" + "assistant" roles only (tool/system not renderable as
/// ChatBubble). Slices the tail so the last HISTORY_LIMIT visible messages
/// are returned in chronological order (oldest first).
#[get("/api/session-messages")]
pub async fn get_session_messages(id: String) -> Result<Vec<ChatMessage>> {
    let state = crate::server::state::global_app_state();
    let all_msgs = state
        .state_store
        .lock()
        .unwrap()
        .get_messages(&id)
        .map_err(|e| ServerFnError::new(format!("get_messages failed: {e}")))?;

    let out: Vec<ChatMessage> = all_msgs
        .iter()
        .filter(|m| m.role == "user" || m.role == "assistant")
        .rev()
        .take(HISTORY_LIMIT)
        .rev()
        .map(|m| ChatMessage {
            role: m.role.clone(),
            content: m.content.clone().unwrap_or_default(),
        })
        .collect();

    Ok(out)
}
```

// NEW private helper — add in the private-helpers section of api.rs
/// Phase 26.7.2 (D-04 / Q6): Extract last message preview for SessionInfo.
/// Prefers last assistant message; falls back to last user message.
/// Skips tool/system roles and None-content entries. Truncates to 80 chars.
fn extract_last_message_preview(msgs: &[ironhermes_state::StoredMessage]) -> Option<String> {
    let preview = msgs.iter().rev()
        .find(|m| m.role == "assistant" && m.content.is_some())
        .or_else(|| msgs.iter().rev().find(|m| m.role == "user" && m.content.is_some()));
    preview
        .and_then(|m| m.content.as_ref())
        .map(|c| c.chars().take(80).collect::<String>())
}
```

// NEW — add to ScreenChat after the auto-scroll use_effect (after line 155)
// Signal-borrow discipline (clippy.toml): session_id_ctx() call-syntax returns
// owned String — no GenerationalRef borrow held across spawn.
// bubbles.write().clear() is synchronous; WriteLock drops at ; before spawn.
use_effect(move || {
    let sid = session_id.read().clone();   // owned String — borrow ends at ;
    if sid == "pending" {
        return;   // D-03/Q3: not yet initialized — skip
    }

    // D-08: always clear on session_id change (clears stale history).
    bubbles.write().clear();
    streaming_id.set(None);   // cancel stale streaming indicator (Q8 recommendation)

    spawn(async move {
        match crate::server::api::get_session_messages(sid).await {
            Ok(msgs) if !msgs.is_empty() => {
                // Map ChatMessage → ChatBubble. next_id read as Copy before
                // any .await (no GenerationalRef across await — clippy.toml).
                let mut id_val = *next_id.read();   // Copy u64 — borrow ends at ;
                let mut history: Vec<ChatBubble> = msgs
                    .into_iter()
                    .filter_map(|m| match m.role.as_str() {
                        "user"      => { let b = ChatBubble::user(id_val, m.content); id_val += 1; Some(b) }
                        "assistant" => { let b = ChatBubble::assistant(id_val, m.content); id_val += 1; Some(b) }
                        _           => None,
                    })
                    .collect();

                // D-02: insert divider after history, before live input area.
                // Guard: only when history is non-empty (Pitfall 3).
                history.push(ChatBubble {
                    id: id_val,
                    kind: ChatBubbleKind::Divider,
                    text: String::new(),
                    tool_rows: vec![],
                });
                id_val += 1;

                // Single write — all history + divider atomically appended.
                // WriteLock acquired and released before next .await (none follows).
                bubbles.write().extend(history);
                next_id.set(id_val);
            }
            _ => {}   // D-07: empty or error → empty stream, no indicator
        }
    });
});
```

**Analog from RESEARCH.md Q5:** The `js_sys::Date::now()` pattern is used in `app_footer.rs` lines 83-98 and `state.rs` lines 292-306. `js-sys` is an unconditional dependency in `Cargo.toml` line 40.

// NEW — add to sessions.rs after the `use` block (after line 22)

/// Phase 26.7.2 (D-05): Format a Unix seconds timestamp as relative time.
/// js_sys::Date::now() returns milliseconds; started_at is seconds (Q7).
/// The #[cfg(not)] stub prevents compile errors on server/native builds.
#[cfg(target_arch = "wasm32")]
fn format_relative(unix_secs_str: &str) -> String {
    let unix_secs: f64 = unix_secs_str.parse().unwrap_or(0.0);
    let now_ms = js_sys::Date::now();
    let then_ms = unix_secs * 1000.0;           // seconds → milliseconds (Pitfall 4)
    let diff_secs = ((now_ms - then_ms) / 1000.0) as i64;
    match diff_secs {
        s if s < 5       => "just now".to_string(),
        s if s < 60      => format!("{}s ago", s),
        s if s < 3_600   => format!("{}m ago", s / 60),
        s if s < 86_400  => format!("{}h ago", s / 3_600),
        s if s < 604_800 => format!("{}d ago", s / 86_400),
        _                => format!("{}w ago", diff_secs / 604_800),
    }
}

#[cfg(not(target_arch = "wasm32"))]
fn format_relative(_: &str) -> String {
    "—".to_string()
}
```
