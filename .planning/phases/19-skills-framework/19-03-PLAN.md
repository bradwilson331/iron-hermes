    ```rust
    fn handle_activate(&self, args: &Value) -> Value {
        let skill_name = args.get("name").and_then(|v| v.as_str()).unwrap_or("");
        let skill = match self.registry.skills.iter().find(|s| s.name == skill_name) {
            Some(s) => s.clone(),
            None => return json!({"status":"error","message": format!("Skill not found: {}", skill_name)}),
        };

        let mut missing_env: Vec<String> = Vec::new();
        let mut missing_creds: Vec<String> = Vec::new();
        let mut setup_help: Option<String> = None;
        let mut required_for_hints: Vec<String> = Vec::new();

        if let Some(meta) = &skill.hermes_metadata {
            for entry in &meta.required_environment_variables {
                if std::env::var(&entry.name).is_err() {
                    missing_env.push(entry.name.clone());
                    if setup_help.is_none() { setup_help = entry.help.clone(); }
                    if let Some(rf) = &entry.required_for { required_for_hints.push(rf.clone()); }
                }
            }
            for entry in &meta.required_credential_files {
                let rel = match entry { CredentialFileEntry::Path(p) => p.clone(), CredentialFileEntry::Structured { path, .. } => path.clone() };
                let abs = self.credential_dir.join(&skill.name).join(&rel);
                if !abs.exists() {
                    missing_creds.push(rel);
                }
            }
        }

        if !missing_env.is_empty() || !missing_creds.is_empty() {
            let mut parts: Vec<String> = Vec::new();
            for e in &missing_env { parts.push(format!("${}", e)); }
            for c in &missing_creds { parts.push(format!("file {}", c)); }
            let suffix = if let Some(first) = required_for_hints.first() { format!(" to {}", first) } else { String::new() };
            let setup_note = format!("I need {}{}.", parts.join(", "), suffix);
            return json!({
                "status": "setup_needed",
                "name": skill.name,
                "readiness_status": "setup_needed",
                "missing_required_environment_variables": missing_env,
                "missing_credential_files": missing_creds,
                "setup_note": setup_note,
                "setup_help": setup_help,
            });
        }

        // Existing success path: read body content, push to active_skills, return {status:ok, name, content}
        // ... (keep existing body-read logic; Plan 04 will inject the config header here)
    }
    ```
