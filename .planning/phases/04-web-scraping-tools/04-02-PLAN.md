```rust
#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_truncate_under_limit() {
        let content = "Short content.";
        let result = truncate_content(content, 100);
        assert_eq!(result, content);
        assert!(!result.contains("[Content truncated"));
    }

    #[test]
    fn test_truncate_at_limit() {
        let content = "Exactly at limit.";
        let result = truncate_content(content, content.chars().count());
        assert_eq!(result, content);
    }

    #[test]
    fn test_truncate_paragraph_break() {
        let content = "First paragraph.\n\nSecond paragraph.\n\nThird paragraph is longer.";
        // Set limit so it falls in the third paragraph
        let result = truncate_content(content, 45);
        assert!(result.starts_with("First paragraph.\n\nSecond paragraph."));
        assert!(result.contains("[Content truncated"));
    }

    #[test]
    fn test_truncate_sentence_break() {
        let content = "First sentence. Second sentence. Third sentence is longer content here.";
        let result = truncate_content(content, 40);
        assert!(result.contains("[Content truncated"));
        // Should cut at a ". " boundary, not mid-word
        assert!(result.starts_with("First sentence. Second sentence.") 
            || result.starts_with("First sentence. Second sentence. "));
    }

    #[test]
    fn test_truncate_word_break() {
        let content = "oneword anotherword thirdword fourthword fifthword";
        let result = truncate_content(content, 25);
        assert!(result.contains("[Content truncated"));
        assert!(!result.contains("thirdword") || result.starts_with("oneword anotherword"));
    }

    #[test]
    fn test_truncate_no_whitespace() {
        let content = "abcdefghijklmnopqrstuvwxyz";
        let result = truncate_content(content, 10);
        assert!(result.contains("[Content truncated at 10 of 26 characters]"));
    }

    #[test]
    fn test_truncate_utf8_emoji() {
        // Each emoji is 4 bytes but 1 char
        let content = "Hello \u{1F600}\u{1F601}\u{1F602}\u{1F603}\u{1F604} world";
        let result = truncate_content(content, 9);
        // Should not panic, should truncate at char boundary
        assert!(result.contains("[Content truncated"));
    }

    #[test]
    fn test_truncate_notice_char_counts() {
        let content = "a]b]c]d]e]f]g]h]i]j]k]l]m]n]o]p]q]r]s]t";
        let total_chars = content.chars().count();
        let result = truncate_content(content, 10);
        assert!(result.contains(&format!("of {} characters]", total_chars)));
    }
}
```
