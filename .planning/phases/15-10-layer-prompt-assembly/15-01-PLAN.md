Wait — the above approach mixes BTreeMap iteration with manual ephemeral slot insertion. Cleaner design: populate ALL slots into the BTreeMap (including ephemeral ones) at build_split() time, then iterate once. The BTreeMap ordering handles everything:

```rust
pub fn build_split(&self) -> (String, String) {
    let mut slots = self.slots.clone();

    // Ensure Identity default (PRMT-03)
    slots.entry(PromptSlot::Identity)
        .or_insert_with(|| DEFAULT_AGENT_IDENTITY.to_string());

    // Slot 2: ToolGuidance — always present
    slots.entry(PromptSlot::ToolGuidance)
        .or_insert_with(|| self.build_tool_guidance());

    // Slot 3: Memory (already in slots if set via load_memory)
    // Slot 4: Skills (already in slots if set via load_skills)
    // Slot 5: ContextFiles (already in slots from load_context)

    // Ephemeral slots — only add if not skip_context_files
    if !self.skip_context_files {
        // Slot 6: Timestamp (always regenerated)
        let ts = self.build_timestamp_block();
        if !ts.is_empty() {
            slots.insert(PromptSlot::Timestamp, ts);
        }

        // Slot 7: PlatformHints
        let ph = self.platform_hint();
        if !ph.is_empty() {
            slots.insert(PromptSlot::PlatformHints, ph);
        }

        // Slot 8: SessionOverlay
        if let Some(ref overlay) = self.active_overlay {
            slots.insert(PromptSlot::SessionOverlay, overlay.clone());
        }
    }

    let mut durable = Vec::new();
    let mut ephemeral = Vec::new();
    for (slot, content) in &slots {
        if *slot == PromptSlot::UserMessage {
            continue; // slot 9 not managed by PromptBuilder
        }
        if slot.is_ephemeral() {
            ephemeral.push(content.clone());
        } else {
            durable.push(content.clone());
        }
    }

    (durable.join("\n\n"), ephemeral.join("\n\n"))
}
```

pub fn load_memory(&mut self) {
    if self.skip_context_files { return; }
    if let Some(ref store) = self.memory_store {
        let store = store.lock().unwrap();
        let mut mem_parts = Vec::new();
        if let Some(block) = store.format_for_system_prompt(MemoryTarget::Memory) {
            mem_parts.push(block);

            mem_parts.push(block);

            self.set_slot(PromptSlot::Memory, mem_parts.join("\n\n"));
        }
    }
}
