    if let Ok(entries) = std::fs::read_dir(&mdc_dir) {
        let mut mdc_parts: Vec<String> = Vec::new();
        for entry in entries.flatten() {
            let path = entry.path();
            if path.extension().and_then(|e| e.to_str()) == Some("mdc") {
                if let Ok(content) = std::fs::read_to_string(&path) {
