```rust
// Intercept internal tools before registry dispatch (D-07, TOOL-04)
match tool_name {
    "memory" => {
        if let Some(ref provider) = self.memory_provider {
            let mut store = provider.lock().unwrap();
            let result = match action {
                // ... delegate to MemoryTool-like logic or store methods directly
            };
            return Ok(result);
        }
    }
    "session_search" => {
        if let Some(ref state) = self.state_store {
            let state_clone = state.clone();
            let args_clone = args.clone();
            let result = tokio::task::spawn_blocking(move || {
                let store = state_clone.lock().unwrap();
                crate::session_search::handle_session_search(&args_clone, &store)
            }).await??;
            return Ok(result);
        }
        return Ok(r#"{"error":"unavailable","reason":"state store not configured"}"#.to_string());
    }
    _ => {}
}
// ... existing registry dispatch continues ...
```

NOTE: The exact interception depends on how the agent loop currently dispatches tools. Read the full agent_loop.rs first. If there is already a memory_provider field, use it. The key requirement is: `memory` and `session_search` tool calls are handled BEFORE registry.execute_tool() and NEVER reach the registry.
