use std::sync::{Arc, Mutex};
use ironhermes_core::{MemoryProvider, MemoryStore, MemoryProviderConfig};
use ironhermes_core::constants::get_hermes_home;

/// Build a memory provider from config. Returns Arc<Mutex<...>> for direct use with MemoryTool.
/// Relocated from ironhermes-core per D-11. Feature-gated providers per D-12.
pub fn build_memory_provider(
    config: &ironhermes_core::config::MemoryConfig,
) -> anyhow::Result<Arc<Mutex<dyn MemoryProvider + Send>>> {
    match config.provider.as_str() {
        "file" => {
            let memory_dir = get_hermes_home().join("memories");
            let store = MemoryStore::new(memory_dir);
            Ok(Arc::new(Mutex::new(store)))
        }
        #[cfg(feature = "memory-sqlite")]
        "sqlite" => {
            let db_path = config.extra.get("db_path")
                .and_then(|v| v.as_str())
                .map(std::path::PathBuf::from)
                .unwrap_or_else(|| get_hermes_home().join("memory.db"));
            let mut provider = memory_sqlite::SqliteMemoryProvider::new(&db_path)?;
            let provider_config = MemoryProviderConfig {
                provider: "sqlite".to_string(),
                memory_dir: get_hermes_home().join("memories"),
                memory_char_limit: ironhermes_core::constants::MEMORY_CHAR_LIMIT,
                user_char_limit: ironhermes_core::constants::USER_CHAR_LIMIT,
                extra: config.extra.clone().unwrap_or_default(),
            };
            // Note: initialize is async but we're in sync context — use block_on or defer
            Ok(Arc::new(Mutex::new(provider)))
        }
        #[cfg(not(feature = "memory-sqlite"))]
        "sqlite" => {
            anyhow::bail!(
                "Memory provider 'sqlite' requires the 'memory-sqlite' feature. \
                 Rebuild with: cargo build --features memory-sqlite"
            );
        }
        #[cfg(not(feature = "memory-duckdb"))]
        "duckdb" => {
            anyhow::bail!(
                "Memory provider 'duckdb' requires the 'memory-duckdb' feature. \
                 Rebuild with: cargo build --features memory-duckdb"
            );
        }
        #[cfg(not(feature = "memory-grafeo"))]
        "grafeo" => {
            anyhow::bail!(
                "Memory provider 'grafeo' requires the 'memory-grafeo' feature. \
                 Rebuild with: cargo build --features memory-grafeo"
            );
        }
        other => {
            anyhow::bail!(
                "Unknown memory provider '{}'. Available providers: file{}",
                other,
                if cfg!(feature = "memory-sqlite") { ", sqlite" } else { "" }
            );
        }
    }
}
```

pub const CREATE_SCHEMA: &str = "
CREATE TABLE IF NOT EXISTS memory_facts (
    id          INTEGER PRIMARY KEY AUTOINCREMENT,
    target      TEXT NOT NULL CHECK(target IN ('memory', 'user')),
    content     TEXT NOT NULL,
    created_at  REAL NOT NULL DEFAULT (julianday('now'))
);

CREATE VIRTUAL TABLE IF NOT EXISTS memory_facts_fts USING fts5(
    content,
    content=memory_facts,
    content_rowid=id
);

CREATE TRIGGER IF NOT EXISTS memory_fts_insert AFTER INSERT ON memory_facts BEGIN
    INSERT INTO memory_facts_fts(rowid, content) VALUES (new.id, new.content);
END;
CREATE TRIGGER IF NOT EXISTS memory_fts_delete AFTER DELETE ON memory_facts BEGIN
    INSERT INTO memory_facts_fts(memory_facts_fts, rowid, content) VALUES('delete', old.id, old.content);
END;
CREATE TRIGGER IF NOT EXISTS memory_fts_update AFTER UPDATE ON memory_facts BEGIN
    INSERT INTO memory_facts_fts(memory_facts_fts, rowid, content) VALUES('delete', old.id, old.content);
    INSERT INTO memory_facts_fts(rowid, content) VALUES (new.id, new.content);
END;
";
```

The implementation mirrors `MemoryStore` logic but uses SQLite instead of files:
- `new(db_path)`: Opens connection, runs CREATE_SCHEMA, sets WAL mode, busy_timeout(5000)
- `initialize()`: No-op (schema created in new())
- `prefetch()`: Loads all entries from memory_facts grouped by target. This IS the frozen snapshot equivalent.
- `sync_turn()`: No-op (mutations are immediate via add/replace/remove)
- `on_session_end()`: No-op
- `shutdown()`: No-op (Connection drops on struct drop)
- `load_from_disk()`: SELECT all from memory_facts, populate internal cache (Vec<String> per target)
- `add(target, content)`: Security scan via `scan_context_content()`. Check capacity (sum of content lengths + delimiters for target). INSERT INTO memory_facts. Return success JSON with chars_used/chars_limit/entries count. Same error format as MemoryStore.
- `replace(target, old_text, new_content)`: Security scan new_content. SELECT entries containing old_text (LIKE '%old_text%'). If 0 matches -> not_found error. If >1 -> ambiguous error. UPDATE the single match. Check capacity after. Same error format.
- `remove(target, old_text)`: SELECT entries containing old_text. Same match logic. DELETE the single match. Return success JSON.
- `format_for_system_prompt(target)`: SELECT all entries for target, compute capacity header same as MemoryStore (with `## Memory (XX% -- N/2,200 chars)` format per D-13), return formatted string. Note: this reads from an internal cache populated at `load_from_disk()` time (frozen snapshot pattern).
- `to_memory_entries()`: Return all cached entries as MemoryEntries.

IMPORTANT: The provider must cache entries at `load_from_disk()` time for the frozen snapshot. Mutations write to SQLite immediately but do NOT update the snapshot cache. `format_for_system_prompt` reads from the snapshot cache. This matches the frozen-snapshot pattern from MemoryStore.

For capacity calculation, replicate the `char_count()` logic: sum of entry lengths + delimiter lengths between them. Use `ENTRY_DELIMITER.len()` from constants.
