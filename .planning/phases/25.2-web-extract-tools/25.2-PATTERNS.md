```rust
async fn fetch_with_firecrawl(url: &str) -> anyhow::Result<String> {
    let api_key = std::env::var("FIRECRAWL_API_KEY")
        .map_err(|_| anyhow::anyhow!("FIRECRAWL_API_KEY not set"))?;
    let timeout_secs = Config::load().map(|c| c.web.timeout_secs).unwrap_or(30);
    debug!("Fetching {} via Firecrawl", url);
    let client = reqwest::Client::builder()
        .timeout(Duration::from_secs(timeout_secs))
        .build()?;
    let response = client
        .post("https://api.firecrawl.dev/v1/scrape")
        .bearer_auth(&api_key)
        .json(&json!({ "url": url, "formats": ["markdown"] }))
        .send().await
        .map_err(|e| anyhow::anyhow!("Firecrawl request failed: {}", e))?;
    // status check ŌåÆ json decode ŌåÆ success check ŌåÆ markdown extraction ŌåÆ title extraction
    // Returns: "# {title}\nSource: {url}\n\n{markdown}" or "Source: {url}\n\n{markdown}"
}
```

```rust
pub async fn extract_youtube(
    url: &str,
    registry: &Arc<ironhermes_core::SkillRegistry>,
) -> anyhow::Result<ExtractionResult> {
    // 1. find the skill record
    let record = registry
        .find("youtube-content")
        .ok_or_else(|| anyhow::anyhow!("youtube-content skill not installed"))?;

    // 2. Invoke the helper script (skills/media/youtube-content/scripts/fetch_transcript.py)
    //    via tokio::process::Command ŌĆö the skill SKILL.md documents:
    //      python3 SKILL_DIR/scripts/fetch_transcript.py "{url}" --text-only
    let script_path = record.path.parent().unwrap().join("scripts/fetch_transcript.py");
    let output = tokio::process::Command::new("python3")
        .arg(&script_path)
        .arg(url)
        .arg("--text-only")
        .output()
        .await
        .map_err(|e| anyhow::anyhow!("youtube-content script failed: {e}"))?;

    // 3. Parse stdout as Markdown; extract title from first "# " heading
    let markdown = String::from_utf8_lossy(&output.stdout).into_owned();
    let title = markdown.lines()
        .find(|l| l.starts_with("# "))
        .map(|l| l.trim_start_matches("# ").to_string())
        .unwrap_or_default();

    Ok(ExtractionResult { url: url.to_string(), title, content: markdown, error: None })
}
```

```rust
pub async fn route_tiers(
    content: &str,
    use_llm_processing: bool,
    config: &ExtractConfig,
    client: &Arc<dyn SummarizationClientHandle>,
    sem: &Arc<Semaphore>,
) -> anyhow::Result<String> {
    let len = content.chars().count();
    match len {
        // Tier 1: < 5_000 ŌĆö return raw (no LLM)
        n if n < config.summary_tier2_threshold_chars => Ok(content.to_string()),
        // Tier 4: > 2_000_000 ŌĆö refuse
        n if n > config.refuse_threshold_chars => {
            anyhow::bail!("content_too_large")

        // Tier 2/3: gated on use_llm_processing
        _ if !use_llm_processing => Ok(truncate_content(content, config_max_chars)),
        // Tier 2: 5KŌĆō500K ŌĆö single-pass summary
        n if n <= config.summary_tier3_threshold_chars => {
            let _permit = sem.acquire().await?;
            client.summarize(SUMMARY_PROMPT.to_string(), content.to_string()).await
        }
        // Tier 3: 500KŌĆō2M ŌĆö chunked
        _ => chunked::summarize_chunked(content, config, client, sem).await,
    }
}
```

### `crates/ironhermes-tools/src/web_extract/summary/chunked.rs` (utility, batch)

**Analog:** `tokio::sync::Semaphore` + `tokio::spawn` ŌĆö no direct in-tree fan-out analog. Pattern derived from Tokio docs + D-15/D-16 decisions.

**NEW PATTERN ŌĆö bounded parallel chunk summarization:**
```rust
use std::sync::Arc;
use tokio::sync::Semaphore;

pub async fn summarize_chunked(
    content: &str,
    config: &ExtractConfig,
    client: &Arc<dyn SummarizationClientHandle>,
    sem: &Arc<Semaphore>,
) -> anyhow::Result<String> {
    // 1. Split into config.summary_chunk_chars segments
    let chunks: Vec<&str> = split_into_chunks(content, config.summary_chunk_chars);
    let total = chunks.len();

    // 2. Parallel chunk summaries ŌĆö acquire semaphore permit per chunk
    let mut handles = Vec::with_capacity(total);
    for (i, chunk) in chunks.iter().enumerate() {
        let sem = sem.clone();
        let client = client.clone();
        let chunk_str = chunk.to_string();
        handles.push(tokio::spawn(async move {
            let _permit = sem.acquire_owned().await?;
            let prompt = format!("Summarize chunk {}/{}: ", i + 1, total);
            client.summarize(prompt, chunk_str).await
        }));
    }

    // 3. Collect chunk summaries
    let mut summaries = Vec::with_capacity(total);
    for (i, handle) in handles.into_iter().enumerate() {
        let s = handle.await??;
        summaries.push(format!("---\n[Chunk {}/{}]\n{}", i + 1, total, s));
    }

    // 4. Synthesis pass ŌĆö single LLM call (D-16)
    let _permit = sem.acquire().await?;
    let joined = summaries.join("\n\n");
    client.summarize(SYNTHESIS_PROMPT.to_string(), joined).await
}
```

```rust
//! Pattern: in-process tool invocation via ToolRegistry + wiremock mock servers.

use std::sync::OnceLock;

use wiremock::matchers::{method, path, header};
use wiremock::{Mock, MockServer, ResponseTemplate};

fn env_lock() -> &'static std::sync::Mutex<()> {
    static LOCK: OnceLock<std::sync::Mutex<()>> = OnceLock::new();
    LOCK.get_or_init(|| std::sync::Mutex::new(()))
}

struct EnvGuard {
    key: &'static str,
    prev: Option<String>,
}
impl EnvGuard {
    fn set(key: &'static str, val: &str) -> Self {
        let prev = std::env::var(key).ok();
        unsafe { std::env::set_var(key, val) };
        Self { key, prev }
    }
}
impl Drop for EnvGuard {
    fn drop(&mut self) {
        unsafe {
            match &self.prev {
                Some(v) => std::env::set_var(self.key, v),
                None => std::env::remove_var(self.key),
            }
        }
    }
}
```
