// — called with skill.name as the argument
let on_toggle = move |name: String| {
    // 1. Capture current optimistic state (Copy bool — no borrow held)
    let current = *toggle_states.read().get(&name).unwrap_or(&false);
    // 2. Optimistic flip
    toggle_states.write().insert(name.clone(), !current);
    // 3. Clear prior error
    toggle_errors.write().remove(&name);
    // 4. Async server call — onclick cannot be async; use spawn
    spawn(async move {
        match crate::server::api::toggle_skill(name.clone()).await {
            Ok(_) => {}  // optimistic state is already correct
            Err(_) => {
                // 5. Revert + set error
                toggle_states.write().insert(name.clone(), current);
                toggle_errors.write().insert(
                    name.clone(),
                    "Toggle failed — try again.".to_string(),
                );
            }
        }
    });
};

#### Full toggle_skill server fn (synthesized from above analogs)

```rust
/// Phase 26.7.3 D-07: Toggle a skill's disabled state.
/// Writes SkillsConfig.disabled (config YAML) and mutates active_skills in-process.
/// Input validation: verify name exists in registry before mutating (RESEARCH Security §).
#[server]
pub async fn toggle_skill(name: String) -> Result<(), ServerFnError> {
    let state = crate::server::state::global_app_state();

    // Validate: reject names not known to the registry (RESEARCH Security §5)
    let record = state.runtime_bundle.skill_registry.find(&name).cloned();

    // 1. Load fresh config from disk — NOT the startup Arc<Config> snapshot
    let mut config = ironhermes_core::config::Config::load()
        .unwrap_or_default();

    // 2. Mutate disabled list (opt-out model: present = disabled, absent = enabled)
    let is_currently_disabled = config.skills.disabled.contains(&name);
    if is_currently_disabled {
        config.skills.disabled.retain(|n| n != &name);  // re-enable
    } else {
        config.skills.disabled.push(name.clone());       // disable
    }

    // 3. Write config back to disk
    config.save()
        .map_err(|e| ServerFnError::new(format!("Config save failed: {e}")))?;

    // 4. Mutate in-process active_skills (std::sync::Mutex — no .await while held)
    // Pattern: api.rs lines 283–287 (active_skills lock)
    let mut skills = state.runtime_bundle.active_skills.lock()
        .map_err(|e| ServerFnError::new(format!("active_skills lock poisoned: {e}")))?;

    if is_currently_disabled {
        // Was disabled → now enabled: push to active_skills if found in registry
        if let Some(rec) = record {
            if !skills.iter().any(|s| s.name == name) {
                skills.push(rec);
            }
        }
    } else {
        // Was enabled → now disabled: remove from active_skills
        skills.retain(|s| s.name != name);
    }
    // skills MutexGuard drops here — no .await held

    Ok(())
}
```

### Signal borrow discipline (clippy-enforced)
**Source:** `crates/iron_hermes_ui/CLAUDE.md` + existing `skills.rs` lines 17–23
**Apply to:** All new signal reads in `ScreenSkills`

Rule: extract all `.read()` guards to owned values before `rsx!`. Owned primitives (`bool`, `usize`, `String`) are safe; `GenerationalRef` must not cross the `rsx!` boundary.

```rust
// SAFE — Copy value extracted before rsx!
let tab_val = active_tab();         // &'static str — Copy
let is_enabled = *toggle_states.read().get(&name).unwrap_or(&false); // bool — Copy, borrow ends at ;

// UNSAFE — ref held across rsx! — DO NOT DO THIS
// let guard = toggle_states.read();  // held open → runtime panic in WASM
```
