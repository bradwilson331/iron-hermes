```rust
fn with_busy_retry<T, F: FnMut() -> Result<T>>(mut f: F) -> Result<T> {
    for attempt in 0u32..3 {
        match f() {
            Ok(v) => return Ok(v),
            Err(StateError::Sqlite(ref e))
                if matches!(e.sqlite_error_code(), Some(rusqlite::ErrorCode::DatabaseBusy))
                    && attempt < 2 =>
            {
                let jitter_ms = 50 + (attempt as u64 * 75); // 50ms, 125ms
                std::thread::sleep(std::time::Duration::from_millis(jitter_ms));
            }
            Err(e) => return Err(e),
        }
    }
    f() // final attempt — propagate error
}
```

Note: `Err(StateError::Sqlite(ref e))` requires matching by reference. Since `StateError` is not `Clone`, the match arm must use `ref` binding. If the borrow checker complains, restructure to extract the error code first: `let code = if let Err(StateError::Sqlite(e)) = &result { e.sqlite_error_code() } else { None };` and branch on that.

Add the `sanitize_fts_query` function (pub, module-level). Per D-05:

```rust
/// Strip FTS5 special operators from user input to prevent query parse errors.
/// Pass `raw: true` in SearchFilter to bypass this.
pub fn sanitize_fts_query(input: &str) -> String {
    let mut result = String::with_capacity(input.len());
    for ch in input.chars() {
        match ch {
            '*' | '^' | '"' | '(' | ')' | '-' | '{' | '}' | ':' => result.push(' '),
            _ => result.push(ch),
        }
    }
    // Remove FTS5 boolean keywords
    let result = result
        .split_whitespace()
        .filter(|w| !matches!(w.to_uppercase().as_str(), "AND" | "OR" | "NOT" | "NEAR"))
        .collect::<Vec<_>>()
        .join(" ");
    if result.trim().is_empty() { String::new() } else { result }
}
```

Replace the existing `search_messages` method with a new version that accepts `SearchFilter`. The old signature `search_messages(&self, query: &str, limit: usize)` becomes `search_messages(&self, filter: &SearchFilter)`. Per D-04 and D-06:

The method builds a dynamic SQL query. When `filter.query` is Some, it uses FTS5 with snippet(). When `filter.query` is None, it does a metadata-only query on the messages table.

```rust
pub fn search_messages(&self, filter: &SearchFilter) -> Result<Vec<SearchResult>> {
    let use_fts = filter.query.is_some();
    let query_text = if let Some(ref q) = filter.query {
        let sanitized = if filter.raw { q.clone() } else { sanitize_fts_query(q) };
        if sanitized.is_empty() { return Ok(vec![]); }
        Some(sanitized)
    } else {
        None
    };

    // Build dynamic WHERE clauses
    let mut conditions: Vec<String> = Vec::new();
    let mut param_values: Vec<Box<dyn rusqlite::types::ToSql>> = Vec::new();
    let mut param_idx = 1usize;

    // Base query depends on whether FTS is used
    let base_select = if use_fts {
        format!(
            "SELECT m.id, m.session_id, m.role, m.content, \
             snippet(messages_fts, 0, '<<', '>>', '...', 32) AS snip, \
             m.timestamp, s.source, s.title \
             FROM messages_fts \
             JOIN messages m ON m.id = messages_fts.rowid \
             JOIN sessions s ON s.id = m.session_id"
        )
    } else {
        format!(
            "SELECT m.id, m.session_id, m.role, m.content, \
             NULL AS snip, \
             m.timestamp, s.source, s.title \
             FROM messages m \
             JOIN sessions s ON s.id = m.session_id"
        )
    };

    if let Some(ref qt) = query_text {
        conditions.push(format!("messages_fts MATCH ?{param_idx}"));
        param_values.push(Box::new(qt.clone()));
        param_idx += 1;
    }
    if let Some(ref src) = filter.source {
        conditions.push(format!("s.source = ?{param_idx}"));
        param_values.push(Box::new(src.clone()));
        param_idx += 1;
    }
    if let Some(ref role) = filter.role {
        conditions.push(format!("m.role = ?{param_idx}"));
        param_values.push(Box::new(role.clone()));
        param_idx += 1;
    }
    if let Some(after) = filter.after {
        conditions.push(format!("m.timestamp >= ?{param_idx}"));
        param_values.push(Box::new(after));
        param_idx += 1;
    }
    if let Some(before) = filter.before {
        conditions.push(format!("m.timestamp <= ?{param_idx}"));
        param_values.push(Box::new(before));
        param_idx += 1;
    }

    let where_clause = if conditions.is_empty() {
        String::new()
    } else {
        format!(" WHERE {}", conditions.join(" AND "))
    };

    let order = if use_fts { "ORDER BY messages_fts.rank" } else { "ORDER BY m.timestamp DESC" };
    let sql = format!("{base_select}{where_clause} {order} LIMIT ?{param_idx}");
    param_values.push(Box::new(filter.limit as i64));

    let params_ref: Vec<&dyn rusqlite::types::ToSql> = param_values.iter().map(|b| b.as_ref()).collect();
    let mut stmt = self.conn.prepare(&sql)?;
    let rows = stmt.query_map(params_ref.as_slice(), |r| {
        Ok(SearchResult {
            message_id: r.get(0)?,
            session_id: r.get(1)?,
            role: r.get(2)?,
            content: r.get(3)?,
            snippet: r.get(4)?,
            context_before: None, // populated below
            context_after: None,  // populated below
            timestamp: r.get(5)?,
            session_source: r.get(6)?,
            session_title: r.get(7)?,
        })
    })?;

    let mut results: Vec<SearchResult> = rows.collect::<rusqlite::Result<Vec<_>>>()?;

    // Populate context_before and context_after (1 message each). Per D-04.
    for result in &mut results {
        // context_before: the message immediately before this one in the same session
        result.context_before = self.conn.query_row(
            "SELECT content FROM messages WHERE session_id = ?1 AND timestamp < ?2 \
             ORDER BY timestamp DESC LIMIT 1",
            params![result.session_id, result.timestamp],
            |r| r.get(0),
        ).optional()?.flatten();

        // context_after: the message immediately after this one in the same session
        result.context_after = self.conn.query_row(
            "SELECT content FROM messages WHERE session_id = ?1 AND timestamp > ?2 \
             ORDER BY timestamp ASC LIMIT 1",
            params![result.session_id, result.timestamp],
            |r| r.get(0),
        ).optional()?.flatten();
    }

    Ok(results)
}
```

IMPORTANT: The `.optional()?.flatten()` pattern works because `query_row` returns `Result<T>`, `.optional()` converts "no rows" `Error` to `Ok(None)`, and `.flatten()` collapses `Option<Option<String>>` to `Option<String>` since content is nullable.
