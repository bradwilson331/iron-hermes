```rust
/// Phase 25 D-18 / TOOL-05: walk every Tool's prerequisites() and prompt the
/// operator for missing required values. Skip option writes to tools.skip_prompts.
pub async fn run_tools_section(
    rl: &mut rustyline::DefaultEditor,
    hermes_home: &Path,
) -> Result<()> {
    // 1. Build a temporary ToolRegistry with all built-in tools.
    let registry = build_full_registry();
    // 2. Query the unavailable list.
    let unavailable = registry.list_unavailable();
    if unavailable.is_empty() {
        println!("All tool prerequisites satisfied — nothing to configure.");
        return Ok(());
    }
    // 3. Walk each tool's missing required prereqs.
    for (tool_name, missing_prereqs) in &unavailable {
        println!();
        println!("Tool: {} ({} unsatisfied required prerequisite{})",
                 tool_name, missing_prereqs.len(),
                 if missing_prereqs.len() == 1 { "" } else { "s" });
        for prereq in missing_prereqs {
            // Display: tool name, prereq.kind, prereq.name, prereq.description
            // Prompt: "Provide value (or type 'skip' to mark this tool never-prompt):"
            // Mask if T-25-02 secret pattern matches.
            let value = prompt_for_prereq_value(rl, tool_name, prereq)?;
            match value {
                PrereqAnswer::Value(v) => apply_prereq_value(hermes_home, prereq, &v)?,
                PrereqAnswer::Skip => apply_skip_prompts(hermes_home, tool_name)?,
                PrereqAnswer::Defer => continue, // operator pressed Enter to skip THIS prompt only
            }
        }
    }
    Ok(())
}
```

```rust
pub async fn run_preflight_check(_cli: &Cli) -> Result<()> {
    let cfg_path = Config::config_path();
    if !cfg_path.exists() {
        return crate::setup::run_setup(None, WizardMode::FirstRun).await;
    }
    match Config::load() {
        Err(_) => crate::setup::run_setup(None, WizardMode::FixMode).await,
        Ok(config) => {
            if !config.validate().is_empty() {
                return crate::setup::run_setup(None, WizardMode::FixMode).await;
            }
            // Phase 25 D-17: tool-prereq probe. Build a registry, call list_unavailable(),
            // emit a stderr banner for required-missing prereqs. NO auto-wizard launch.
            let registry = crate::setup::build_full_registry(); // shared with run_tools_section
            // Apply config.tools.skip_prompts filter so suppressed tools don't banner.
            let unavailable = registry.list_unavailable();
            let skip: std::collections::HashSet<&str> = config.tools.skip_prompts
                .iter().map(|s| s.as_str()).collect();
            let active: Vec<_> = unavailable.into_iter()
                .filter(|(n, _)| !skip.contains(n.as_str()))
                .collect();
            if !active.is_empty() {
                eprintln!("⚠ Tool prerequisites unsatisfied — run `hermes toolset setup` to configure:");
                for (tool, missing) in &active {
                    let prereq_names: Vec<_> = missing.iter().map(|p| p.name.as_str()).collect();
                    eprintln!("  - {} ({})", tool, prereq_names.join(", "));
                }
            }
            Ok(())
        }
    }
}
```
