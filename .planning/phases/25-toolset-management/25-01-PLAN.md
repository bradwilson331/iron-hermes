// AFTER Plan 1 (additive — signature of every method UNCHANGED):
pub struct Prerequisite {
    pub kind: String,        // "env_var" | "config_field"
    pub name: String,        // "FIRECRAWL_API_KEY" or "search.brave_api_key"
    pub description: String,
    pub required: bool,      // true = blocks; false = optional/fallback
}

#[async_trait]
pub trait Tool: Send + Sync {
    fn name(&self) -> &str;
    fn toolset(&self) -> &str;
    fn description(&self) -> &str;
    fn schema(&self) -> ToolSchema;

    fn is_available(&self) -> bool {
        // Default: walk prerequisites(), return true iff every required env_var prereq is satisfied.
        self.prerequisites()
            .iter()
            .filter(|p| p.required)
            .all(|p| match p.kind.as_str() {
                "env_var" => std::env::var(&p.name).is_ok(),
                "config_field" => true, // checked at config load site, not at trait level
                _ => true,
            })
    }

    fn prerequisites(&self) -> Vec<Prerequisite> {
        vec![]
    }

    async fn execute(&self, args: serde_json::Value) -> anyhow::Result<String>;
}
