connects via stdio and HTTP transports using the rmcp SDK, discovers tools, registers them
as Box<dyn Tool> into the ToolRegistry via register_dynamic, handles reconnection with
exponential backoff, and supports reload (disconnect all, re-discover, re-register).

external MCP servers, discover their tools, and make them available to the LLM as if they

    ```rust
    use async_trait::async_trait;
    use ironhermes_core::ToolSchema;
    use ironhermes_tools::Tool;
    use tokio::sync::{mpsc, oneshot};

    /// Request sent from McpTool::execute to the server task's dispatch loop.
    pub struct McpCallRequest {
        pub tool_name: String,  // original MCP tool name (not prefixed)
        pub arguments: serde_json::Value,
        pub response_tx: oneshot::Sender<anyhow::Result<String>>,
    }

    pub struct McpTool {
        prefixed_name: String,     // "filesystem__read_file" (D-06)
        server_name: String,       // "filesystem"
        original_name: String,     // "read_file"
        description: String,       // "[MCP: filesystem] Read a file at path" (D-11)
        schema: ToolSchema,
        call_tx: mpsc::Sender<McpCallRequest>,
    }

    impl McpTool {
        pub fn new(
            server_name: &str,
            original_name: &str,
            original_description: &str,
            input_schema: serde_json::Value,
            call_tx: mpsc::Sender<McpCallRequest>,
        ) -> Self {
            let prefixed_name = make_prefixed_name(server_name, original_name);
            let description = format!("[MCP: {server_name}] {original_description}");
            let schema = ToolSchema::new(&prefixed_name, &description, input_schema);
            Self { prefixed_name, server_name: server_name.to_string(),
                   original_name: original_name.to_string(), description, schema, call_tx }
        }
    }

    /// D-06: server__tool naming with sanitization (hyphens and dots -> underscores).
    pub fn make_prefixed_name(server_name: &str, tool_name: &str) -> String {
        let safe_server = server_name.replace('-', "_").replace('.', "_");
        let safe_tool = tool_name.replace('-', "_").replace('.', "_");
        format!("{safe_server}__{safe_tool}")
    }

    #[async_trait]
    impl Tool for McpTool {
        fn name(&self) -> &str { &self.prefixed_name }
        fn toolset(&self) -> &str { "mcp" }
        fn description(&self) -> &str { &self.description }
        fn schema(&self) -> ToolSchema { self.schema.clone() }
        async fn execute(&self, args: serde_json::Value) -> anyhow::Result<String> {
            let (resp_tx, resp_rx) = oneshot::channel();
            self.call_tx.send(McpCallRequest {
                tool_name: self.original_name.clone(),
                arguments: args,
                response_tx: resp_tx,
            }).await.map_err(|_| anyhow::anyhow!("MCP server '{}' disconnected", self.server_name))?;
            match resp_rx.await {
                Ok(result) => result.map_err(|e| anyhow::anyhow!("{}", crate::security::sanitize_error(&e.to_string()))),
                Err(_) => Err(anyhow::anyhow!("MCP server '{}' dropped request", self.server_name)),
            }
        }
    }
    ```

    ```rust
    use crate::config::McpServerConfig;
    use crate::security::build_safe_env;

    /// Connect to a stdio MCP server. Returns the running service.
    /// D-19: env_clear + build_safe_env for security.
    pub async fn connect_stdio(config: &McpServerConfig) -> Result<rmcp::service::RunningService<rmcp::RoleClient, ()>> {
        let command = config.command.as_ref()
            .ok_or_else(|| anyhow::anyhow!("stdio transport requires 'command' field"))?;
        let safe_env = build_safe_env(&config.env);
        use rmcp::transport::ConfigureCommandExt;
        let args = config.args.clone();
        let transport = rmcp::transport::TokioChildProcess::new(
            tokio::process::Command::new(command).configure(move |cmd| {
                for arg in &args { cmd.arg(arg); }
                cmd.env_clear();
                for (k, v) in &safe_env { cmd.env(k, v); }
            })
        )?;
        use rmcp::ServiceExt;
        let client = ().serve(transport).await?;
        Ok(client)
    }

    /// Connect to an HTTP/StreamableHTTP MCP server.
    pub async fn connect_http(config: &McpServerConfig) -> Result<rmcp::service::RunningService<rmcp::RoleClient, ()>> {
        let url = config.url.as_ref()
            .ok_or_else(|| anyhow::anyhow!("HTTP transport requires 'url' field"))?;
        let transport = rmcp::transport::StreamableHttpClientTransport::from_uri(url);
        use rmcp::ServiceExt;
        let client = ().serve(transport).await?;
        Ok(client)
    }
    ```

    ```rust
    use std::sync::atomic::{AtomicU32, Ordering};

    /// Callback type for sampling: given a prompt, return the LLM completion.
    pub type SamplingCallback = Arc<dyn Fn(String, Option<String>) -> futures::future::BoxFuture<'static, anyhow::Result<String>> + Send + Sync>;

    /// Rate-limited sampling handler.
    pub struct SamplingHandler {
        callback: SamplingCallback,
        calls_this_minute: AtomicU32,
        max_rpm: u32,
        max_tool_rounds: u32,
    }

    impl SamplingHandler {
        pub fn new(callback: SamplingCallback, max_rpm: u32, max_tool_rounds: u32) -> Self {
            Self { callback, calls_this_minute: AtomicU32::new(0), max_rpm, max_tool_rounds }
        }

        pub async fn handle_sampling_request(&self, prompt: String, model_hint: Option<String>) -> anyhow::Result<String> {
            let count = self.calls_this_minute.fetch_add(1, Ordering::SeqCst);
            if count >= self.max_rpm {
                return Err(anyhow::anyhow!("Sampling rate limit exceeded ({} rpm)", self.max_rpm));
            }
            (self.callback)(prompt, model_hint).await
        }
    }
    ```

    **1. Create crates/ironhermes-mcp/src/server_task.rs:**

    Per-server tokio task with reconnection logic (D-04, D-05):

    ```rust
    use crate::config::McpServerConfig;
    use crate::tool::{McpTool, McpCallRequest};
    use crate::transport;
    use crate::security::sanitize_error;
    use ironhermes_tools::ToolRegistry;
    use std::sync::Arc;
    use std::time::Duration;
    use tokio::sync::{mpsc, RwLock};
    use tokio_util::sync::CancellationToken;

    const MAX_RETRIES: u32 = 5;
    const MAX_BACKOFF: Duration = Duration::from_secs(60);

    pub struct ServerTaskResult {
        pub tool_names: Vec<String>,  // prefixed names of registered tools
    }

    /// Run a single MCP server's lifecycle: connect, discover tools, register,
    /// serve tool calls, reconnect on failure with exponential backoff.
    pub async fn run_server_task(
        name: String,
        mut config: McpServerConfig,
        registry: Arc<RwLock<ToolRegistry>>,
        cancel_token: CancellationToken,
    ) -> ServerTaskResult {
        // Interpolate env vars in config
        crate::config::interpolate_config(&mut config);

        let mut registered_names: Vec<String> = Vec::new();
        let mut retries = 0u32;
        let mut backoff = Duration::from_secs(1);

        loop {
            match connect_and_serve(&name, &config, &registry, &cancel_token).await {
                Ok(names) => {
                    registered_names = names;
                    break; // clean shutdown
                }
                Err(_) if cancel_token.is_cancelled() => break,
                Err(e) => {
                    retries += 1;
                    if retries > MAX_RETRIES {
                        tracing::warn!(server = %name, error = %sanitize_error(&e.to_string()),
                            "MCP server exhausted {MAX_RETRIES} retries");
                        break;
                    }
                    tracing::warn!(server = %name, retry = retries, max = MAX_RETRIES,
                        backoff_secs = backoff.as_secs(),
                        error = %sanitize_error(&e.to_string()),
                        "MCP server disconnected, retrying");
                    tokio::select! {
                        _ = tokio::time::sleep(backoff) => {}
                        _ = cancel_token.cancelled() => break,
                    }
                    backoff = (backoff * 2).min(MAX_BACKOFF);
                }
            }
        }
        ServerTaskResult { tool_names: registered_names }
    }

    async fn connect_and_serve(
        name: &str,
        config: &McpServerConfig,
        registry: &Arc<RwLock<ToolRegistry>>,
        cancel_token: &CancellationToken,
    ) -> anyhow::Result<Vec<String>> {
        // Connect via appropriate transport
        let client = if config.command.is_some() {
            transport::connect_stdio(config).await?
        } else if config.url.is_some() {
            transport::connect_http(config).await?
        } else {
            return Err(anyhow::anyhow!("Server '{}' has neither 'command' nor 'url'", name));
        };

        // Discover tools
        let mcp_tools = client.list_all_tools().await?;
        tracing::info!(server = %name, tool_count = mcp_tools.len(), "MCP tools discovered");

        // Create dispatch channel
        let (call_tx, mut call_rx) = mpsc::channel::<McpCallRequest>(32);

        // Register tools into ToolRegistry (D-06, D-08, D-11)
        let mut registered_names = Vec::new();
        {
            let mut guard = registry.write().await;
            for mcp_tool in &mcp_tools {
                let tool_name = mcp_tool.name.as_str();

                // D-08: enabled_tools filtering
                if let Some(ref enabled) = config.enabled_tools {
                    if !enabled.iter().any(|e| e == tool_name) {
                        continue;
                    }
                }

                let description = mcp_tool.description.as_deref().unwrap_or("");
                let input_schema = mcp_tool.input_schema.clone();
                // Convert input_schema from rmcp type to serde_json::Value
                let schema_value = serde_json::to_value(&input_schema).unwrap_or_default();

                let tool = McpTool::new(name, tool_name, description, schema_value, call_tx.clone());
                let prefixed = tool.name().to_string();
                guard.register_dynamic(Box::new(tool));
                registered_names.push(prefixed);
            }
        } // write guard dropped

        tracing::info!(server = %name, registered = registered_names.len(), "MCP tools registered");

        // Serve tool call requests until cancellation
        loop {
            tokio::select! {
                _ = cancel_token.cancelled() => {
                    break;
                }
                request = call_rx.recv() => {
                    match request {
                        Some(req) => {
                            // Dispatch to MCP server
                            let timeout = Duration::from_secs(config.timeout);
                            let result = tokio::time::timeout(timeout, async {
                                // Call the MCP tool via rmcp client
                                let call_result = client.call_tool(rmcp::model::CallToolRequestParam {
                                    name: req.tool_name.into(),
                                    arguments: Some(serde_json::from_value(req.arguments).unwrap_or_default()),
                                }).await;
                                match call_result {
                                    Ok(response) => {
                                        // Extract text content from response
                                        let text = response.content.iter()
                                            .filter_map(|c| {
                                                if let rmcp::model::Content::Text(t) = c {
                                                    Some(t.text.as_str())
                                                } else { None }
                                            })
                                            .collect::<Vec<_>>()
                                            .join("\n");
                                        if response.is_error.unwrap_or(false) {
                                            Err(anyhow::anyhow!("{}", sanitize_error(&text)))
                                        } else {
                                            Ok(text)
                                        }
                                    }
                                    Err(e) => Err(anyhow::anyhow!("{}", sanitize_error(&e.to_string()))),
                                }
                            }).await;

                            let final_result = match result {
                                Ok(inner) => inner,
                                Err(_) => Err(anyhow::anyhow!("MCP tool call timed out after {}s", config.timeout)),
                            };
                            let _ = req.response_tx.send(final_result);
                        }
                        None => break, // all senders dropped
                    }
                }
            }
        }

        Ok(registered_names)
    }
    ```

    - `client.call_tool(param)` -> tool result with content array

    ```rust
    use crate::config::McpServerConfig;
    use crate::server_task;
    use ironhermes_tools::ToolRegistry;
    use std::collections::HashMap;
    use std::sync::Arc;
    use tokio::sync::RwLock;
    use tokio::task::JoinHandle;
    use tokio_util::sync::CancellationToken;

    /// Manages all MCP server connections and their tool registrations.
    pub struct McpManager {
        registry: Arc<RwLock<ToolRegistry>>,
        tasks: tokio::sync::Mutex<HashMap<String, (JoinHandle<server_task::ServerTaskResult>, CancellationToken)>>,
        configs: tokio::sync::Mutex<HashMap<String, McpServerConfig>>,
    }

    impl McpManager {
        pub fn new(registry: Arc<RwLock<ToolRegistry>>) -> Self {
            Self {
                registry,
                tasks: tokio::sync::Mutex::new(HashMap::new()),
                configs: tokio::sync::Mutex::new(HashMap::new()),
            }
        }

        /// Start all configured MCP servers. Call during agent startup.
        /// D-07: one-shot discovery at startup.
        pub async fn start_all(&self, configs: HashMap<String, McpServerConfig>) {
            let mut tasks = self.tasks.lock().await;
            let mut stored_configs = self.configs.lock().await;

            for (name, config) in configs {
                if !config.enabled {
                    tracing::info!(server = %name, "MCP server disabled, skipping");
                    continue;
                }
                let cancel = CancellationToken::new();
                let handle = tokio::spawn(server_task::run_server_task(
                    name.clone(),
                    config.clone(),
                    self.registry.clone(),
                    cancel.clone(),
                ));
                tasks.insert(name.clone(), (handle, cancel));
                stored_configs.insert(name, config);
            }
        }

        /// Shutdown all running server tasks.
        pub async fn shutdown_all(&self) {
            let mut tasks = self.tasks.lock().await;
            for (name, (handle, cancel)) in tasks.drain() {
                tracing::info!(server = %name, "Shutting down MCP server");
                cancel.cancel();
                let _ = handle.await;
                // Unregister tools for this server
                let mut guard = self.registry.write().await;
                let removed = guard.unregister_by_prefix(&name);
                tracing::debug!(server = %name, removed, "Unregistered MCP tools");
            }
        }

        /// D-09/D-12: Reload all MCP servers.
        /// Disconnects all, re-reads config, reconnects, re-discovers.
        pub async fn reload(&self, new_configs: HashMap<String, McpServerConfig>) {
            self.shutdown_all().await;
            self.start_all(new_configs).await;
        }

        /// Return names of servers that have active tasks.
        pub fn connected_server_names(&self) -> Vec<String> {
            // Use try_lock since this may be called from sync context
            if let Ok(tasks) = self.tasks.try_lock() {
                tasks.keys().cloned().collect()
            } else {
                Vec::new()
            }
        }

        /// Count of registered MCP tools in the registry.
        pub async fn registered_tool_count(&self) -> usize {
            let guard = self.registry.read().await;
            guard.list_tools().iter().filter(|name| name.contains("__")).count()
        }
    }

    impl Drop for McpManager {
        fn drop(&mut self) {
            // Cancel all tasks on drop
            if let Ok(tasks) = self.tasks.try_lock() {
                for (_, (_, cancel)) in tasks.iter() {
                    cancel.cancel();
                }
            }
        }
    }
    ```
