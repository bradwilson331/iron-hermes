fn write_skill_dir(dir: &Path, files: &[(&str, &[u8])]) -> std::io::Result<()> {
    for (rel, bytes) in files {
        let path = dir.join(rel);
        if let Some(parent) = path.parent() {
            std::fs::create_dir_all(parent)?;
        }
        std::fs::write(&path, bytes)?;
    }
    Ok(())
}

        ("references/note.md", b"# Notes\n"),
    ]).unwrap();
    let identifier = format!("local:{}", source_dir.path().to_str().unwrap());

    // Snapshot source dir BEFORE install
    let before = snapshot_dir(source_dir.path());

    // Install
    let rc = cmd_install(&cfg, &identifier, false).await.unwrap();
    assert_eq!(rc, 0);

    // Snapshot source dir AFTER install (must be unchanged — install copies, not moves)
    let after_install = snapshot_dir(source_dir.path());
    assert_eq!(before, after_install, "install must not modify source dir");

    // Remove
    let rc = cmd_remove(&cfg, "my-skill").await.unwrap();  // name from SKILL.md frontmatter
    assert_eq!(rc, 0);

    // Snapshot source dir AFTER remove (must STILL be byte-identical)
    let after_remove = snapshot_dir(source_dir.path());
    assert_eq!(before, after_remove,
        "T-21.8.1-07: cmd_remove must not touch the source directory under any circumstances");

    // Install dir under skills_root MUST be gone
    let install_path = hermes_home.path().join("skills/my-skill");  // adjust per actual category layout
    // Or: walk skills_root and assert empty
    let still_present = any_file_named(&hermes_home.path().join("skills"), "SKILL.md");
    assert!(!still_present, "install dir must be cleaned up");
}

fn snapshot_dir(dir: &Path) -> Vec<(String, String)> {
    // Recursive walk; return Vec<(rel_path, sha256_hex)> sorted by path
    // Hand-rolled per RULE 1; use sha2 (already a workspace dep)
    let mut out = Vec::new();
