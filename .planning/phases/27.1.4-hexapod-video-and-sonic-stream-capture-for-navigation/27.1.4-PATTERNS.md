#### New `stream_distance` execute arm

Model directly on `read_distance` arm (lines 369–378) extended into a loop:

```rust
"stream_distance" => {
    let samples = args["samples"].as_i64().unwrap_or(5)
        .clamp(1, STREAM_DISTANCE_MAX_SAMPLES) as usize;
    let mut readings: Vec<i64> = Vec::with_capacity(samples);

    for i in 0..samples {
        let read_fut = send_and_read_line(&addr, DISTANCE_CMD);
        let outcome = timeout(Duration::from_secs(3), read_fut).await;
        let raw = map_read_outcome(outcome, &addr)?;
        if raw.starts_with("Error:") {
            return Ok(raw);
        }
        // Parse numeric value from CMD_SONIC#<dist>\n — NOT parse_distance_response()
        // which returns a formatted string (RESEARCH Pitfall 6)
        let parts: Vec<&str> = raw.trim().split('#').collect();
        let dist = parts.get(1)
            .and_then(|s| s.trim().parse::<i64>().ok())
            .unwrap_or(0);
        readings.push(dist);
        if i + 1 < samples {
            tokio::time::sleep(Duration::from_millis(200)).await;  // D-20: tokio sleep
        }
    }

    let min = readings.iter().copied().min().unwrap_or(0);
    let max = readings.iter().copied().max().unwrap_or(0);
    let avg = readings.iter().copied().sum::<i64>() as f64 / readings.len() as f64;
    let list: Vec<String> = readings.iter().map(|d| d.to_string()).collect();
    Ok(format!("Distances: [{}] cm | min={} max={} avg={:.1}", list.join(", "), min, max, avg))
}
```

## Shared Patterns

### Stateless TCP fire-and-forget

**Source:** `crates/ironhermes-tools/src/hexapod_tcp.rs` lines 167–175
**Apply to:** `camera_pan` and `camera_tilt` arms in `hexapod_tcp.rs`

```rust
async fn send_fire_and_forget(addr: &str, cmd: &str) -> anyhow::Result<()> {
    let mut stream = TcpStream::connect(addr).await?;
    stream.write_all(cmd.as_bytes()).await?;
    stream.flush().await?;
    Ok(())
}
```
