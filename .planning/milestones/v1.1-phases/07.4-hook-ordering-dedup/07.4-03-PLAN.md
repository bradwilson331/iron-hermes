    #[tokio::test]
    async fn test_telegram_round_trip_exactly_one_message_received() {
        // D-09 / roadmap SC#2: a single Telegram message must produce exactly one
        // MessageReceived and exactly one ResponseSent in a shared HookRegistry.
        //
        // Pre-07.4: agent_loop.rs:131 and :200 fired both kinds with platform="agent" and
        // empty chat_id, producing 2x of each event per real message. Plan 07.4-02
        // removed those fires. This test is the mechanical guard that a future edit to
        // agent_loop.rs, run_agent, or a downstream helper cannot reintroduce the

        // (the same shape as test_cron_tick_exactly_one_message_received in runner.rs)
        // rather than standing up a full mock PlatformAdapter + mock LlmClient +
        // GatewayMessageHandler pipeline. The direct-drive form matches the existing
        // cron test pattern and keeps the test deterministic and fast. If a richer
        // integration harness is added later (e.g. a `TestHandler::run_agent_with_mocks`
        // helper), this test can be upgraded in place.

        let mut registry = HookRegistry::new(HooksConfig::default());
        let captured: Arc<std::sync::Mutex<Vec<HookEvent>>> =
            Arc::new(std::sync::Mutex::new(Vec::new()));
        let cap = captured.clone();

            cap.lock().unwrap().push(event);
        }));
        let registry = Arc::new(registry);

        //   - :217-229  MessageReceived { platform="telegram", chat_id=event.chat_id, ... }
        //   - :385-397  ResponseSent    { platform="telegram", chat_id=event.chat_id, ... }

        // for the message-received fire and a separate one for the response-sent fire
        // in the current code — BUT D-02 of phase 06 specifies per-turn correlation, so
        // this test asserts the EXACTLY-ONE invariant, not the same-request-id invariant
        // (which is separately tracked in Phase 06 tech debt).
        let req_id_received = uuid::Uuid::new_v4().to_string();
        let req_id_sent = uuid::Uuid::new_v4().to_string();

            &req_id_received,

                platform: "telegram".to_string(),
                chat_id: "tg-chat-42".to_string(),
                content_preview: "hello".to_string(),

                response_preview: "hi there".to_string(),
